32  /**
33   * This class renders a standard WRA and loads a specific named association and places it into the model.
34   * <p/>
35   * It exists as a sample that a company can use to build its own page actions.
36   *
37   * @author Tony Field
38   * @since 2012-08-24
39   */
40  public class GeneralRenderPage implements Action {
41  
42      protected static final Log LOG = LogFactory.getLog(GeneralRenderPage.class.getPackage().getName());
43  
44      /**
45       * Inject an ICS into the action for convenience
46       */
47      @InjectForRequest
48      public ICS ics;
49  
50      /**
51       * Provide a place to put the data that will be passed into the page context for use with expression language
52       */
53      @InjectForRequest
54      public Model model;
55  
56      /**
57       * Provide a DAO that allows an asset to be easily mapped
58       */
59      @InjectForRequest
60      protected TemplateAssetAccess assetDao;
61  
62      /**
63       * Process and handle the request. This method might be responsible for invoking
64       * the view as well.
65       *
66       * @param ics Content Server context.
67       */
68      public void handleRequest(final ICS ics) {
69          // get the asset id corresponding to the ICS variables c, cid
70          AssetId id = assetDao.currentId();
71  
72          // read the asset into a format that is easy to manipulate
73          TemplateAsset asset = assetDao.read(id);
74  
75          // this would be a very good place for a security check or something similar!
76  
77          // load the current asset into a TemplateAsset object (not actually used in this example)
78          model.add("asset", new ScatteredAsset(asset.getDelegate()));
79  
80          // put the named associations for this asset into the model as well for convenience
81          for (AssetAssociationDef assoc : asset.getAssetTypeDef().getAssociations()) {
82              String assocName = assoc.getName();
83              if (assoc.isMultiple()) {
84                  model.add(assocName, asset.getAssociatedAssets(assocName));
85              } else {
86                  model.add(assocName, asset.getAssociatedAsset(assocName));
87              }
88          }
89      }
90  }
