38                      ics.StreamHeader("Location", location);
39                      ics.StreamHeader(RenderPageAdapter.STATUS_HEADER,
40                              Integer.toString(HttpServletResponse.SC_MOVED_TEMPORARILY));
41  
42                  } else {
43                      super.handleRequest(ics);
44                  }
45                  break;
