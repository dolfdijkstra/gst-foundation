18   * A sample on how to manipulate the childpagename in an Seed, to be used in an XML Wrapper element.
19   * 
20   * @author Dolf.Dijkstra
21   * @since 25 jun. 2012

23  public class DetectorSeed implements Seed2 {
24  
25      private static final String CHILDPAGENAME = "childpagename";
26  
27      static DeviceDetector detector = new MobiForgeDeviceDetector();
28  
29      private ICS ics;
30  
31      /*
32       * (non-Javadoc)
33       * 
34       * @see
35       * COM.FutureTense.XML.Template.Seed#Execute(COM.FutureTense.Interfaces.
36       * FTValList, COM.FutureTense.Interfaces.FTValList)
37       */
38      @Override
39      public String Execute(FTValList in, FTValList out) {
40  
41      	DeviceType type = detector.detectDeviceType(ics);
42  
43          String childpagename = ics.GetVar(CHILDPAGENAME);
44          if (StringUtils.isNotBlank(childpagename)) {
45              if (type == DeviceType.MOBILE) {
46                  ics.SetVar(CHILDPAGENAME, childpagename + "_mobile");
47              } else if (type == DeviceType.TABLET) {
48                  ics.SetVar(CHILDPAGENAME, childpagename + "_tablet");
49              }
50          }
51          return null;
52      }

61      @Override

63          ics = ips.GetICSObject();
64      }
