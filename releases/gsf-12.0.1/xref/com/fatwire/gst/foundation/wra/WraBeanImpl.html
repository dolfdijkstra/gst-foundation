16  package com.fatwire.gst.foundation.wra;
17  
18  import java.util.Date;
19  
20  import com.fatwire.assetapi.data.AssetId;
21  import com.fatwire.gst.foundation.facade.assetapi.asset.TemplateAsset;
22  
23  /**
24   * Simple WRA bean
25   *
26   * @author Tony Field
27   * @since Jul 21, 2010
28   * 
29   * 
30   * @deprecated as of release 12.x, replace with WCS 12c's native vanity URLs support.
31   * 
32   */
33  public class WraBeanImpl implements WebReferenceableAsset {
34      private AssetId id;
35      private String name;
36      private String description;
37      private String subtype;
38      private String status;
39      private String metaTitle;
40      private String metaDescription;
41      private String metaKeyword;
42      private String h1Title;
43      private String linkText;
44      private String path;
45      private String template;
46      private Date startDate;
47      private Date endDate;
48  
49      public WraBeanImpl() {
50      }
51  
52      public WraBeanImpl(TemplateAsset asset) {
53          id = asset.getAssetId();
54          name = asset.asString("name");
55          description = asset.asString("description");
56          subtype = asset.asString("subtype");
57          status = asset.asString("status");
58          metaTitle = asset.asString("metatitle");
59          metaDescription = asset.asString("metadescription");
60          metaKeyword = asset.asString("metakeyword");
61          h1Title = asset.asString("h1title");
62          linkText = asset.asString("linktext");
63          path = asset.asString("path");
64          template = asset.asString("template");
65          startDate = asset.asDate("startdate");
66          endDate = asset.asDate("enddate");
67      }
68  
69      public WraBeanImpl(WebReferenceableAsset wra) {
70          id = wra.getId();
71          name = wra.getName();
72          description = wra.getDescription();
73          subtype = wra.getSubtype();
74          status = wra.getStatus();
75          metaTitle = wra.getMetaTitle();
76          metaDescription = wra.getMetaDescription();
77          metaKeyword = wra.getMetaKeyword();
78          h1Title = wra.getH1Title();
79          linkText = wra.getLinkText();
80          path = wra.getPath();
81          template = wra.getTemplate();
82          startDate = wra.getStartDate();
83          endDate = wra.getEndDate();
84      }
85  
86      public AssetId getId() {
87          return id;
88      }
89  
90      public void setId(AssetId id) {
91          this.id = id;
92      }
93  
94      public String getName() {
95          return name;
96      }
97  
98      public void setName(String name) {
99          this.name = name;
100     }
101 
102     public String getDescription() {
103         return description;
104     }
105 
106     public void setDescription(String description) {
107         this.description = description;
108     }
109 
110     public String getSubtype() {
111         return subtype;
112     }
113 
114     public void setSubtype(String subtype) {
115         this.subtype = subtype;
116     }
117 
118     public String getStatus() {
119         return status;
120     }
121 
122     public void setStatus(String status) {
123         this.status = status;
124     }
125 
126     public String getMetaTitle() {
127         return metaTitle;
128     }
129 
130     public void setMetaTitle(String metaTitle) {
131         this.metaTitle = metaTitle;
132     }
133 
134     public String getMetaDescription() {
135         return metaDescription;
136     }
137 
138     public void setMetaDescription(String metaDescription) {
139         this.metaDescription = metaDescription;
140     }
141 
142     public String getMetaKeyword() {
143         return metaKeyword;
144     }
145 
146     public void setMetaKeyword(String metaKeyword) {
147         this.metaKeyword = metaKeyword;
148     }
149 
150     public String getH1Title() {
151         return h1Title;
152     }
153 
154     public void setH1Title(String h1Title) {
155         this.h1Title = h1Title;
156     }
157 
158     public String getLinkText() {
159         return linkText;
160     }
161 
162     public void setLinkText(String linkText) {
163         this.linkText = linkText;
164     }
165 
166     public String getPath() {
167         return path;
168     }
169 
170     public void setPath(String path) {
171         this.path = path;
172     }
173 
174     public String getTemplate() {
175         return template;
176     }
177 
178     public void setTemplate(String template) {
179         this.template = template;
180     }
181 
182     public Date getStartDate() {
183         return startDate;
184     }
185 
186     public void setStartDate(Date startDate) {
187         this.startDate = startDate;
188     }
189 
190     public Date getEndDate() {
191         return endDate;
192     }
193 
194     public void setEndDate(Date endDate) {
195         this.endDate = endDate;
196     }
197 
198     public String toString() {
199         return (getId() != null) ? getId().toString() : "[null id]";
200     }
201 }
