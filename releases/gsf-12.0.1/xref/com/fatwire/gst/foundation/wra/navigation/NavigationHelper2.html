48  /**
49   * Used to retrieve the Navigation Bar data. See the description of
50   * {@link #getSitePlan(String)} for more details.
51   * 
52   * 
53   * @author Dolf Dijkstra
54   * @since Jun 8, 2012
55   * 
56   * @deprecated as of release 12.x, will be replaced with a brand new, significantly improved NavigationService implementation (coming soon)
57   */
58  public class NavigationHelper2 extends AbstractNavigationHelper {
59      /**
60       * Constructor. Initializes assetEffectiveDate to null.
61       * 
62       * @param ics object
63       */
64      public NavigationHelper2(final ICS ics) {
65          super(ics);
66  
67      }
68  
69      public NavigationHelper2(ICS ics, TemplateAssetAccess assetTemplate, WraCoreFieldDao wraDao,
70              AliasCoreFieldDao aliasDao) {
71          super(ics, assetTemplate, wraDao, aliasDao);
72      }
73  
74      /**
75       * Called from public {@link #getSitePlan(int, AssetId)}. See that
76       * function's description for details
77       * 
78       * @param depth the depth of the tree to retrieve, -1 for unlimited depth.
79       * @param pageId id of the page asset
80       * @param level starting level number when traversing the site plan tree
81       * @return NavNode of the site plan tree
82       */
83      @Override
84      protected NavigationNode getSitePlan(final int depth, final AssetId pageId, final int level,
85              final DimensionFilterInstance dimensionFilter) {
86          // check the start/end date of the page asset
87          LogDep.logDep(ics, pageId);
88          if (!isValidOnDate(ics, pageId, assetEffectiveDate)) {
89              // the input object is not valid. Abort
90              if (LOG.isDebugEnabled()) {
91                  LOG.debug("Input asset " + pageId + " is not effective on " + assetEffectiveDate);
92              }
93              return null;
94          }
95  
96          // determine if it's a wra, a placeholder or an alias
97  
98          final AssetData pageData = AssetDataUtils.getAssetData(ics, pageId, "subtype", "name");
99          final String subtype = AttributeDataUtils.asString(pageData.getAttributeData("subtype"));
100         final String name = AttributeDataUtils.asString(pageData.getAttributeData("name"));
101         final boolean isNavigationPlaceholder = NAVBAR_NAME.equals(subtype);
102         final NavigationNode node = new NavigationNode();
103 
104         node.setPage(pageId);
105         node.setLevel(level);
106         node.setPagesubtype(subtype);
107         node.setPagename(name);
108 
109         if (isNavigationPlaceholder) {
110             // no link if it's just a placeholder
111         } else if (NAVBAR_LINK.equals(subtype)) {
112             // add the WRA asset to the node
113             final AssetClosure closure = new NodeWraAssetClosure(node);
114             final DateFilterClosure dateFilterClosure = new DateFilterClosure(PreviewContext.getPreviewDate(ics,
115                     assetEffectiveDate), closure);
116 
117             // retrieve the unnamed association(s) based on date filter
118             if (dimensionFilter == null) {
119                 assetTemplate.readAssociatedAssets(pageId, "-", dateFilterClosure);
120             } else {
121 
122                 final Collection<AssetId> associatedWraList = assetTemplate.readAssociatedAssetIds(pageId, "-");
123                 for (final AssetId child : dimensionFilter.filterAssets(associatedWraList)) {
124                     assetTemplate.readAsset(child, dateFilterClosure);
125                 }
126             }
127 
128         } else {
129             // not a GSTNavLink, probably 11g page
130             final AssetClosure closure = new NodeNoneWraAssetClosure(node, "linktext");
131             final DateFilterClosure dateFilterClosure = new DateFilterClosure(PreviewContext.getPreviewDate(ics,
132                     assetEffectiveDate), closure);
133 
134             // retrieve the unnamed association(s) based on date filter
135             if (dimensionFilter == null) {
136                 assetTemplate.readAsset(pageId, dateFilterClosure);
137             } else {
138                 for (final AssetId child : dimensionFilter.filterAssets(Collections.singleton(pageId))) {
139                     assetTemplate.readAsset(child, dateFilterClosure);
140                 }
141             }
142 
143         }
144 
145         if (depth < 0 || depth > level) {
146             // get the children in the Site Plan
147             final List<AssetId> childrenIDs = ListPages.getChildPages(ics, pageId.getId());
148             for (final AssetId aid : childrenIDs) {
149                 // note recursing here
150                 final NavigationNode kidInfo = getSitePlan(depth, aid, level + 1, dimensionFilter);
151                 if (kidInfo != null && kidInfo.getPage() != null) {
152                     node.addChild(kidInfo);
153                 }
154             }
155         }
156         return node;
157     }

159     class NodeWraAssetClosure implements AssetClosure {
160 
161         private final NavigationNode node;
162 
163         public NodeWraAssetClosure(final NavigationNode node) {
164             this.node = node;
165         }
166 
167         @Override
168         public boolean work(final AssetData asset) {
169             final AssetId id = asset.getAssetId();
170             node.setId(id);
171             if (isGstAlias(id)) {
172                 decorateAsAlias(id, node);
173             } else {
174                 decorateAsWra(id, node);
175             }
176             return false; // needs to return only one node
177         }
178 
179         protected void decorateAsAlias(final AssetId id, final NavigationNode node) {
180             final Alias alias = aliasDao.getAlias(id);
181             final String url = getUrlForAlias(alias);
182             final String linktext = alias.getLinkText();
183             if (url != null) {
184                 node.setUrl(url);
185             }
186             if (linktext != null) {
187                 node.setLinktext(linktext);
188             }
189 
190         }
191 
192         protected void decorateAsWra(final AssetId id, final NavigationNode node) {
193 
194             final WebReferenceableAsset wra = wraDao.getWra(id);
195             final String url = getUrlForWra(wra);
196             final String linktext = wra.getLinkText();
197             if (url != null) {
198                 node.setUrl(url);
199             }
200             if (linktext != null) {
201                 node.setLinktext(linktext);
202             }
203         }
204 
205         /**
206          * Get the URL for the alias.
207          * 
208          * For external links, the targeturl attribute is rendered.
209          * 
210          * For Aliases that refer to another WRA, the alias is allowed to
211          * override any WRA fields. For instance, the path, and the template can
212          * be overridden by an alias for a WRA (though the template in the Alias
213          * had better be typeless, or a template of the same name must exist in
214          * the WRA's asset type or there will be a problem).
215          * 
216          * @param alias Alias bean, which of course is also a WRA.
217          * @return url
218          */
219         protected String getUrlForAlias(final Alias alias) {
220             if (alias.getTargetUrl() != null) {
221                 return alias.getTargetUrl();
222             } else {
223                 return getUrlForWra(alias);
224             }
225         }
226 
227         /**
228          * Get the URL to use for the web-referenceable asset.
229          * 
230          * @param wra WebReferenceableAsset bean
231          * @return url
232          */
233         protected String getUrlForWra(final VanityAsset wra) {
234             if (wra.getTemplate() == null || wra.getTemplate().length() == 0) {
235                 LOG.warn("Asset " + wra + " does not have a valid template set.");
236                 return null;
237             }
238             String wrapper = ics.GetProperty("com.fatwire.gst.foundation.url.wrapathassembler.dispatcher",
239                     "ServletRequest.properties", true);
240             if (!Utilities.goodString(wrapper)) {
241                 wrapper = "GST/Dispatcher";
242             }
243             return new WraUriBuilder(wra, wrapper).toURI(ics);
244 
245         }
246 
247     }

249     class NodeNoneWraAssetClosure implements AssetClosure {
250 
251         private final NavigationNode node;
252         private final String linkTextAttribute;
253 
254         public NodeNoneWraAssetClosure(final NavigationNode node, String linkTextAttribute) {
255             this.node = node;
256             this.linkTextAttribute = linkTextAttribute;
257 
258         }
259 
260         @Override
261         public boolean work(final AssetData asset) {
262             final AssetId id = asset.getAssetId();
263             node.setId(id);
264             decorateAsNoneWra(id, node);
265             return false; // needs to return only one node
266         }
267 
268         protected void decorateAsNoneWra(final AssetId id, final NavigationNode node) {
269             TemplateAsset asset = assetTemplate.read(id);
270             final String url = getUrl(asset);
271 
272             final String linktext = asset.asString(linkTextAttribute);
273 
274             if (url != null) {
275                 node.setUrl(url);
276             }
277             if (linktext != null) {
278                 node.setLinktext(linktext);
279             }
280         }
281 
282         private String getUrl(TemplateAsset asset) {
283             String template = asset.asString("template");
284             if (StringUtils.isNotBlank(template)) {
285                 LOG.warn("Asset " + asset.getAssetId() + " does not have a valid template set.");
286                 return null;
287             }
288             String wrapper = ics.GetProperty("com.fatwire.gst.foundation.url.wrapathassembler.dispatcher",
289                     "ServletRequest.properties", true);
290             if (!Utilities.goodString(wrapper)) {
291                 wrapper = "GST/Dispatcher";
292             }
293             return new WraUriBuilder(asset.getAssetId()).wrapper(wrapper).template(template).toURI(ics);
294 
295         }

297     }
