34  /**
35   * 
36   * This class' only purpose is to save time when testing legacy (and deprecated) Actions
37   * in WCS 12c. We don't want to create an entire (gsf-sample-legacy) module just for this,
38   * let alone rewire the std MAVEN build strategy to fit this in.
39   * 
40   * In future releases, we will be implementing brand new samples for the CORE features / classes,
41   * but those will go into the gsf-sample module. 
42   * 
43   * Once all action-related classes are physically wiped out from the GSF project's codebase,
44   * we will get rid of this class, too.
45   * 
46   * @author fvillalba
47   *
48   */
49  public class LegacyType2Action implements Action {
50  	
51  	private static final Logger LOG = LoggerFactory.getLogger("tools.gsf.samples.LegacyType2Action");
52  
53  	/**
54       * Inject an ICS into the action for convenience
55       */
56      @InjectForRequest
57      public ICS ics;
58   
59      /**
60       * Provide a place to put the data that will be passed into the page context for use with expression language
61       */
62      @InjectForRequest
63      public Model model;
64   
65      /**
66       * Provide a DAO that allows an asset to be easily mapped
67       */
68      @InjectForRequest
69      protected TemplateAssetAccess templateAssetAccess;
70      
71      /**
72       * Bind rendermode to local variable
73       */
74      @Bind(value="rendermode")
75      protected String myRenderMode;
76      
77      /**
78       * Bind map key to local variable
79       */
80      @Mapping(value="myMappedAsset")
81      protected String mappedAsset;
82   
83      public void handleRequest(ICS ics) {
84          // get the asset id corresponding to the ICS variables c, cid
85          AssetId id = templateAssetAccess.currentId();
86          
87          TemplateAsset asset = templateAssetAccess.read(id);
88  		model.add("legacyGsfAsset", new ScatteredAsset(asset.getDelegate()));
89  		
90  		LOG.info("Value copied from ICS variable rendermode onto local variable myRenderMode = " + myRenderMode);
91  		
92  		LOG.info("Value copied from Map Key myMappedAsset onto local variable mappedAsset = " + mappedAsset);
93      }
94  }
