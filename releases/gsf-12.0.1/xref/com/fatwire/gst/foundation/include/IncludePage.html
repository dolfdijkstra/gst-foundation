19  import java.util.Collections;
20  import java.util.HashMap;
21  import java.util.List;
22  import java.util.Map;
23  import java.util.Map.Entry;
24  
25  import org.apache.commons.lang3.StringUtils;
26  
27  import COM.FutureTense.ContentServer.PageData;
28  import COM.FutureTense.Interfaces.FTValList;
29  import COM.FutureTense.Interfaces.ICS;
30  
31  import com.fatwire.gst.foundation.facade.runtag.render.CallTemplate.Style;
32  import com.fatwire.gst.foundation.facade.runtag.render.ContentServer;
33  import com.fatwire.gst.foundation.facade.runtag.render.SatellitePage;

35  /**
36   * Class that calls render:contentserver or render:satellitepage based on style.
37   * 
38   * @author Dolf Dijkstra
39   * @since Apr 13, 2011
40   * 
41   * @deprecated as of release 12.x, replace with OOTB features (e.g. callelement tag, calltemplate tag, ics.RunTag and the like)
42   * 
43   */
44  public class IncludePage implements Include {
45      public static final List<String> FORBIDDEN_VARS = Collections.unmodifiableList(Arrays.asList("tid", "eid", "seid",
46              "packedargs", "variant", "context", "pagename", "rendermode", "ft_ss"));
47  
48      private Style style;
49      private final Map<String, String> list = new HashMap<String, String>();
50      private final String pagename;
51      private String packedArgs;
52  
53      private final List<String> pc;
54      private final ICS ics;
55  
56      /**
57       * @param ics Content Server context object
58       * @param pagename page name
59       */
60      public IncludePage(final ICS ics, final String pagename) {
61          this(ics, pagename, Style.embedded);
62      }
63  
64      /**
65       * @param ics Content Server context object
66       * @param pagename page name
67       * @param style string value of style
68       */
69      public IncludePage(final ICS ics, final String pagename, final Style style) {
70          this.ics = ics;
71          this.style = style;
72          this.pagename = pagename;
73          final String[] keys = ics.pageCriteriaKeys(pagename);
74          if (keys == null) {
75              throw new IllegalArgumentException("Can't find page criteria for " + pagename
76                      + ". Please check if pagecriteria are set for " + pagename + ".");
77          }
78          pc = Arrays.asList(keys);
79      }
80  
81      /*
82       * (non-Javadoc)
83       * 
84       * @see
85       * com.fatwire.gst.foundation.include.Include#include(COM.FutureTense.Interfaces
86       * .ICS)
87       */
88      public void include(final ICS ics) {
89          switch (style) {
90              case embedded: {
91                  final ContentServer tag = new ContentServer(pagename);
92                  for (final Entry<String, String> e : list.entrySet()) {
93                      tag.setArgument(e.getKey(), e.getValue());
94                  }
95                  if (StringUtils.isNotBlank(packedArgs)) {
96                      tag.setArgument("PACKEDARGS", packedArgs);
97                  }
98                  final String s = tag.execute(ics);
99                  if (s != null) {
100                     ics.StreamText(s);
101                 }
102             }
103                 break;
104             case pagelet: {
105                 final SatellitePage tag = new SatellitePage(pagename);
106                 for (final Entry<String, String> e : list.entrySet()) {
107                     tag.setArgument(e.getKey(), e.getValue());
108                 }
109                 if (StringUtils.isNotBlank(packedArgs)) {
110                     tag.setPackedArgs(packedArgs);
111                 }
112 
113                 final String s = tag.execute(ics);
114                 if (s != null) {
115                     ics.StreamText(s);
116                 }
117             }
118                 break;
119             case element: {
120                 PageData data = ics.getPageData(pagename);
121                 if (!data.isRegistered())
122                     throw new IllegalArgumentException(pagename + " is not a registered page.");
123                 String element = data.getRootElement();
124                 FTValList ftv = argsToFTValList();
125                 ics.CallElement(element, ftv);
126             }
127             default:
128                 throw new IllegalStateException("Can't handle style " + style);
129         }
130 
131     }
132 
133     /**
134      * @return
135      */
136     @SuppressWarnings("unchecked")
137     private FTValList argsToFTValList() {
138         FTValList ftv = new FTValList();
139         ftv.putAll(list);
140         return ftv;
141     }
142 
143     /**
144      * @param name argument name
145      * @param value argument value
146      * @return this
147      * @see "com.fatwire.gst.foundation.facade.runtag.render.CallTemplate#setArgument(java.lang.String, java.lang.String)"
148      */
149     public IncludePage argument(final String name, final String value) {
150         if (FORBIDDEN_VARS.contains(name)) {
151             throw new IllegalArgumentException("Can't deal with " + name);
152         }
153 
154         if (pc.contains(name)) {
155             list.put(name, value);
156         } else {
157             throw new IllegalArgumentException( name + " is not part of the page criteria: " + pc.toString());
158         }
159         return this;
160     }
161 
162     /**
163      * Adds packedargs.
164      * 
165      * @param s string value of packed arguments
166      * @return this
167      * @see "com.fatwire.gst.foundation.facade.runtag.render.CallTemplate#setPackedargs(java.lang.String)"
168      */
169     public IncludePage packedargs(final String s) {
170         this.packedArgs = s;
171         return this;
172     }
173 
174     /**
175      * Copies the ics variables identified by the name array
176      * 
177      * @param name array of argument names
178      * @return this
179      */
180     public IncludePage copyArguments(final String... name) {
181         if (name == null) {
182             return this;
183         }
184         for (final String n : name) {
185             argument(n, ics.GetVar(n));
186         }
187         return this;
188     }
189 
190     public IncludePage style(final Style s) {
191         style = s;
192         return this;
193     }
194 
195     /**
196      * Sets Style to embedded
197      * 
198      * @return this
199      */
200     public IncludePage embedded() {
201         return style(Style.embedded);
202     }
203 
204     /**
205      * Sets Style to pagelet
206      * 
207      * @return this
208      */
209     public IncludePage pagelet() {
210         return style(Style.pagelet);
211     }
