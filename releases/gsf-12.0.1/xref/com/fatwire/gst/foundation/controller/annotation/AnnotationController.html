33  /**
34   * The Controller delegates the business logic to an annotated method. In this
35   * method the special logic for the request is executed. Method selection is
36   * done via the {@literal @}IcsVariable annotation. This works of the premise that a special
37   * name/value pair on the ics scope set before this controller is executed, is
38   * enough to determine the selection for the business logic method.
39   * 
40   * @author Dolf Dijkstra
41   * @since Apr 15, 2011
42   * @see IcsVariable
43   * 
44   * 
45   * @deprecated as of release 12.x
46   * 
47   */
48  public abstract class AnnotationController extends AbstractController {
49  
50      public AnnotationController(ICS ics) {
51          super(ics);
52      }
53  
54      /*
55       * (non-Javadoc)
56       * 
57       * @see "com.fatwire.gst.foundation.controller.AbstractController#doExecute()"
58       */
59      @Override
60      protected void doExecute() {
61  
62          // record seid and eid and any other deps required
63          RenderUtils.recordBaseCompositionalDependencies(ics);
64          final ControllerMappingResolver resolver = new ControllerMappingResolver();
65          final Object target = getTarget();
66          final Method m = resolver.findControllerMethod(ics, target);
67          if (m == null) {
68              executeDefault();
69          } else {
70              invokeControllerMethod(m, target);
71          }
72      }
73  
74      /**
75       * @param m method to invoke
76       * @param target target object
77       */
78      protected void invokeControllerMethod(final Method m, final Object target) {
79          try {
80              m.invoke(target, new Object[] { ics });
81          } catch (final IllegalArgumentException e) {
82              handleException(e);
83          } catch (final IllegalAccessException e) {
84              handleException(e);
85          } catch (final InvocationTargetException e) {
86              handleException(e);
87          }
88  
89      }
90  
91      /**
92       * @return this
93       */
94      protected Object getTarget() {
95          return this;
96      }
97  
98      /**
99       * Default operation to execute if no matching annotated method can be
100      * found.
101      */
102     abstract void executeDefault();
103 
104     @Override
105     protected void handleException(final Exception e) {
106         if (e instanceof CSRuntimeException) {
107             handleCSRuntimeException((CSRuntimeException) e);
108         } else {
109             sendError(500, e);
110         }
111     }
112 
113     /**
114      * This method transforms errno values into http status codes and sets them
115      * using the X-Fatwire-Status header.
116      * <p>
117      * Only some errnos are handled by this base class.
118      * <p>
119      * More info coming soon
120      * 
121      * @param e exception
122      */
123     protected void handleCSRuntimeException(final CSRuntimeException e) {
124         switch (e.getErrno()) {
125             case 400:
126             case ftErrors.badparams:
127                 sendError(400, e);
128                 break;
129             case 404:
130             case ftErrors.pagenotfound:
131                 sendError(404, e);
132                 break;
133             case 403:
134             case ftErrors.noprivs:
135                 sendError(403, e);
136                 break;
137             default:
138                 sendError(500, e);
139                 break;
140         }
141     }
142 
143     /**
144      * Record compositional dependencies that are required for the controller
145      */
146     protected void recordCompositionalDependencies() {
147         if (ics.isCacheable(ics.GetVar(ftMessage.PageName))) {
148             if (goodString(ics.GetVar("seid"))) {
149                 LogDep.logDep(ics, "SiteEntry", ics.GetVar("seid"));
150             }
151             if (goodString(ics.GetVar("eid"))) {
152                 LogDep.logDep(ics, "CSElement", ics.GetVar("eid"));
153             }
154         }
155     }
