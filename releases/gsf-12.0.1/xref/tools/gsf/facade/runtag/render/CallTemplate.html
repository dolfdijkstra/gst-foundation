32  /**
33   * CallTemplate tag with many improvements around context and style.
34   * <p>
35   * <code>
36   * &lt;RENDER.CALLTEMPLATE SITE="site name"
37   * SLOTNAME="name of slot"
38   * TID="caller Template or CSElement id" [TTYPE="caller Template or CSElement"]
39   * [C="asset type"] [CID="asset id"] [TNAME="target Template or CSElement name"]
40   * [CONTEXT="context override"] [STYLE="pagelet or element"]
41   * [VARIANT="template variant name"] [PACKEDARGS="packed arguments"]&gt;
42   * [&lt;RENDER.ARGUMENT NAME="variable1" VALUE="value1"/&gt;]
43   * &lt;/RENDER.CALLTEMPLATE&gt;
44   * </code>
45   * </p>
46   * 
47   * <p>
48   * <b>
49   * MAIN CHANGES WITH REGARDS TO LEGACY render:calltemplate FACADE:
50   * - "override" property / mechanism has been removed from this tag.
51   * - Default "style" property / mechanism has been removed from this tag.
52   * - Style calculation intelligence has been removed from this tag. Style must now be explicitly set by the caller; otherwise, this facade will throw an Exception.
53   * </b>
54   * </p>
55   *
56   * @author Tony Field
57   * @author Dolf Dijkstra
58   * @author Freddy Villalba
59   * @since Aug 15, 2016
60   */
61  public class CallTemplate extends TagRunnerWithRenderArguments {

71      private String site, type, tname, cid;
72      private Style style;

75          element, pagelet, embedded
76      }

79          Template, CSElement
80      }

82      public CallTemplate() {
83          super("RENDER.CALLTEMPLATE");
84      }

86      /**
87       * Sets up CallTemplate.
88       * 
89       * <b>IMPORTANT: note that Style is now required, as opposed to previous versions of the GSF. Also, bear in mind that,
90       * in WCS 12c, using style="element" implies the called Template's Controller does NOT get invoked at all.</b> 
91       *
92       * @param slotname slot name
93       * @param tname    template name
94       * @param type     template type
95       * @param style    call style (e.g. pagelet, embedded or element) 
96       */
97      public CallTemplate(final String slotname, final String tname, final Type type, final CallTemplate.Style style) {
98          super("RENDER.CALLTEMPLATE");
99          setSlotname(slotname);
100         setTname(tname);
101         setTtype(type);
102         setStyle(style);
103         setContext("");
104     }
105 
106     /**
107      * Checks the current settings and based on the current and target template
108      * state set the style to a best guess. This is only done if the developer
109      * didnot explicitly set the style.
110      */
111     @Override
112     protected void preExecute(final ICS ics) {
113         // consider to set site to ics.GetVar("site"); as helper for reasonable
114 
115         // default value
116         readConfig(ics);
117     	
118     	// if style has not been explicitly set, bomb out
119     	if (this.style == null) {
120     		throw new IllegalStateException("Starting GSF-12, you must explicitly set 'style'. Also, bear in mind that, in WCS 12c, calling a template with style='element' prevents the corresponding Controller from getting invoked.");
121     	}
122         
123         ics.ClearErrno();
124         super.preExecute(ics);
125     }
126 
127     @Override
128     protected void postExecute(ICS ics) {
129         site = null;
130         type = null;
131         tname = null;
132         cid = null;
133         style = null;
134         super.postExecute(ics);
135     }

192     void readConfig(final ICS ics) {
193         if (configLoaded) {
194             return;
195         }
196         CallTemplate.config_FixPageCriteria = "true".equals(getProperty(ics, "config_FixPageCriteria"));
197         CallTemplate.configLoaded = true;
198     }
199 
200     private String getProperty(final ICS ics, final String name) {
201         String val = System.getProperty(CallTemplate.class.getName() + "." + name);
202         if (!Utilities.goodString(val)) {
203             val = ics.GetProperty(CallTemplate.class.getName() + "." + name, "futuretense_xcel.ini", true);
204         }
205         LOG.trace(CallTemplate.class.getName() + "." + name + "=" + val);
206         return val;
207     }

213     protected void handleError(ICS ics) {
214 
215         int errno = ics.GetErrno();
216         if (errno != -10004) // error checking was too aggressive, any error set
217         // by an element would leak into this handler
218         {
219             return;
220         }
221         FTValList arguments = list;
222         ftErrors complexError = ics.getComplexError();
223         String pagename = ics.GetVar("pagename");
224         String elementname = ics.ResolveVariables("CS.elementname");
225         String msg = "ics.runTag(RENDER.CALLTEMPLATE) failed for tname: " + list.getValString("TNAME") + " for asset: "
226                 + list.getValString("C") + ":" + list.getValString("CID") + " within pagename:" + pagename;
227         if (elementname != null) {
228             msg += " and element:" + elementname;
229         }
230         msg += ".";
231         ics.ClearErrno();
232         throw new TagRunnerRuntimeException(msg, errno, arguments, complexError, pagename, elementname);
233     }
