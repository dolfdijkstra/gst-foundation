25  public class CallTemplateTest extends TestCase {
26  
27      public void testFixPageCriteria() {
28          x();
29      }
30  
31      void x() {
32          String[] pc = {"locale", "p"};
33          Map<String, Object> m = new HashMap<String, Object>();
34  
35          m.put("TTYPE", "foo");
36          m.put("ARGS_locale", "en");
37          m.put("ARGS_foo", "foo");
38          m.put("ARGS_TNAME", "foo");
39          for (final Iterator<Entry<String, Object>> i = m.entrySet().iterator(); i.hasNext(); ) {
40              final Entry<String, ?> e = i.next();
41              final String key = e.getKey();
42              // only inspect arguments that start with ARGS_
43              if (key.startsWith(CallTemplate.ARGS)) {
44  
45                  String shortKey = key.substring(CallTemplate.ARGS.length());
46                  boolean found = CallTemplate.CALLTEMPLATE_EXCLUDE_VARS.contains(shortKey);
47                  //System.out.println(key +" " + found);
48                  if (!found) {
49                      for (final String c : pc) {
50                          if (c.equalsIgnoreCase(shortKey)) {
51                              found = true;
52                              break;
53                          }
54                      }
55                  }
56                  if (!found) {
57                      //System.out.println("Argument '" + key + "' not found as PageCriterium " + ". Arguments are: " + m.keySet().toString() + ". PageCriteria: " + Arrays.asList(pc));
58                      // we could correct this by calling as an element
59                      // or by removing the argument
60  
61                      i.remove();
62                      //System.out.println("Argument '" + key + "' is removed from the call as it is not a PageCriterium.");
63                  }
64  
65              }
66          }
67      }
