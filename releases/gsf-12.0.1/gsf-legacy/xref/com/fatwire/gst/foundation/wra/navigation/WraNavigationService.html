96      public WraNavigationService(ICS ics, TemplateAssetAccess assetTemplate, AliasCoreFieldDao aliasDao,
97              final DimensionFilterInstance dimensionFilter, Date previewDate) {
98          super(ics, assetTemplate, "linktext", "path");
99          this.aliasDao = aliasDao;
100         this.dimensionFilter = dimensionFilter;
101         this.date = previewDate;

105     @Override
106     protected NavigationNode getNode(Row row, int level, int depth, String linkAttribute) {
107         long nid = row.getLong("nid");
108         long pageId = row.getLong("oid");
109 
110         AssetId pid = assetTemplate.createAssetId(row.getString("otype"), pageId);
111         if (!isValidOnDate(ics, pid, date)) {
112             // the input object is not valid. Abort
113             if (LOG.isDebugEnabled()) {
114                 LOG.debug("Input asset " + pid + " is not effective.");
115             }
116             return null;
117         }
118         LogDep.logDep(ics, pid); // probably redundant call
119         TemplateAsset asset = assetTemplate.read(pid, "name", "subtype", "template", pathAttribute, linkAttribute);
120 
121         final NavigationNode node = new NavigationNode();
122 
123         node.setPage(pid);
124         node.setLevel(level);
125         node.setPagesubtype(asset.getSubtype());
126         node.setPagename(asset.asString("name"));
127         final boolean isNavigationPlaceholder = NAVBAR_NAME.equals(asset.asString("subtype"));
128         final boolean isNavigationLink = NAVBAR_LINK.equals(asset.asString("subtype"));
129         if (isNavigationPlaceholder) {
130             // return node without an associated asset
131         } else if (isNavigationLink) {
132 
133             Collection<AssetId> assocs;
134             assocs = assetTemplate.readAssociatedAssetIds(pid, "-");
135             if (dimensionFilter != null)
136                 assocs = dimensionFilter.filterAssets(assocs);
137             for (AssetId assoc : assocs) {
138                 if (isValidOnDate(ics, assoc, date)) {
139                     if (isGstAlias(assoc)) {
140                         final Alias alias = aliasDao.getAlias(assoc);
141                         node.setId(alias.getId());
142 
143                         final String url = alias.getTargetUrl() != null ? alias.getTargetUrl() : getUrl(alias
144                                 .getTarget());
145                         final String linktext = alias.getLinkText();
146 
147                         if (url != null) {
148                             node.setUrl(url);
149                         }
150                         if (linktext != null) {
151                             node.setLinktext(linktext);
152                         }
153                     } else {
154                         node.setId(assoc);
155                         asset = assetTemplate.read(assoc, "name", "subtype", "template", pathAttribute, linkAttribute);
156                         final String url = getUrl(asset);
157 
158                         if (url != null) {
159                             node.setUrl(url); // escape by default.
160                         }
161 
162                         final String linktext = asset.asString(linkAttribute);
163 
164                         if (linktext != null) {
165                             node.setLinktext(linktext);
166                         } else {
167                             node.setLinktext(asset.asString("name"));
168                         }
169                     }
170                 }
171             }
172 
173         } else {
174             // TODO Add support for locale (date checking is already done at
175             // start of this function).
176 
177             // other subtype
178             final String url = getUrl(asset);
179 
180             if (url != null) {
181                 node.setUrl(url); // escape by default.
182             }
183 
184             final String linktext = asset.asString(linkAttribute);
185 
186             if (linktext != null) {
187                 node.setLinktext(linktext);
188             } else {
189                 node.setLinktext(asset.asString("name"));
190             }
191 
192         }
193 
194         if (depth < 0 || depth > level) {
195             // get the children in the Site Plan, note recursing here
196             Collection<NavigationNode> children = getNodeChildren(nid, level + 1, depth, linkAttribute);
197             for (final NavigationNode kid : children) {
198                 if (kid != null && kid.getPage() != null) {
199                     node.addChild(kid);
200                 }
201             }
202         }
203         return node;
204 
205     }
206 
207     private String getUrl(AssetId assoc) {
208         TemplateAsset asset = assetTemplate.read(assoc, "name", "subtype", "template", pathAttribute);
209         return getUrl(asset);
210     }
211 
212     @Override
213     protected Collection<NavigationNode> getNodeChildren(final long nodeId, final int level, final int depth,
214             String linkAttribute) {
215         StatementParam param = CHILD_STMT.newParam();
216         param.setLong(0, nodeId);
217 
218         IListIterable root = SqlHelper.select(ics, CHILD_STMT, param);
219         List<NavigationNode> collection = new LinkedList<NavigationNode>();
220         for (Row row : root) {
221             final NavigationNode node = getNode(row, level, depth, linkAttribute);
222             if (node != null)
223                 collection.add(node);
224 
225         }
226         return collection;
227     }

229     /**
230      * Builds up a URI for this asset, using the pathAttribute and the template
231      * field of the asset
232      * 
233      * @param asset template asset
234      * @return the uri, xml escaped
235      */
236     protected String getUrl(TemplateAsset asset) {
237         String template = asset.asString("template");
238         String path = asset.asString(pathAttribute);
239         if (StringUtils.isBlank(template)) {
240             LOG.debug("Asset " + asset.getAssetId() + " does not have a valid template set.");
241             return null;
242         }
243         if (StringUtils.isBlank(path)) {
244             LOG.debug("Asset " + asset.getAssetId()
245                     + " does not have a valid path set. Defaulting to a non Vanity Url.");
246             return new TemplateUriBuilder(asset.getAssetId(), template).toURI(ics);
247         }
248 
249         String wrapper = ics.GetProperty("com.fatwire.gst.foundation.url.wrapathassembler.dispatcher",
250                 "ServletRequest.properties", true);
251         if (!Utilities.goodString(wrapper)) {
252             wrapper = "GST/Dispatcher";
253         }
254         String uri = new WraUriBuilder(asset.getAssetId()).wrapper(wrapper).template(template).toURI(ics);
255         return StringEscapeUtils.escapeXml(uri);
256     }

258     /**
259      * Return true if the asset type is a GSTAlias asset type. May be overridden
260      * if customers are attempting to retrofit this class for alias-like
261      * functionality that is not implemented by the GSTAlias asset type.
262      * 
263      * @param id asset for which a link is required
264      * @return true if the asset is an alias, false if it is a web-referenceable
265      *         asset
266      */
267     protected boolean isGstAlias(final AssetId id) {
268         return GST_ALIAS_TYPE.equals(id.getType());
269     }
