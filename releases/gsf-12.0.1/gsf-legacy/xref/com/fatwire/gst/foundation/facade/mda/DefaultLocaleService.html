138     @Override
139     public AssetId findTranslation(final AssetId id, final long preferredDimension, final long dimensionSetId) {
140         if (id == null) {
141             throw new IllegalArgumentException("Required Asset ID missing");
142         }
143 
144         if (_isInputAssetDimensionPreferred(id, preferredDimension)) {
145             LOG.debug("Input dimension is already in the preferred dimension.  Not invoking dimension set filter.  Asset: "
146                     + id + ", dimension: " + preferredDimension);
147             return id;
148         } else {
149             LOG.debug("About to look for translations.  Input asset id: " + id + ", dimension set: " + dimensionSetId
150                     + ", preferred dimension: " + preferredDimension);
151         }
152 
153         // *****************************************************************************
154         // The core business logic of this helper class is encapsulated in these
155         // 3 lines
156         final DimensionSetInstance dimset = _getDimensionSet(dimensionSetId);
157         return findTranslation(id, preferredDimension, dimset);
158 
159     }

169     @Override
170     public AssetId findTranslation(final AssetId id, final long preferredDimension, final String dimensionSetName) {
171         if (id == null) {
172             throw new IllegalArgumentException("Required Asset ID missing");
173         }
174         final Dimension locale = getLocaleForAsset(id);
175 
176         if (locale == null) {
177             LOG.debug("Asset is not localized.  Not invoking dimension set filter.  Asset: " + id);
178             return id;
179         }
180         if (locale.getId().getId() == preferredDimension) {
181             LOG.debug("Input dimension is already in the preferred dimension.  Not invoking dimension set filter.  Asset: "
182                     + id + ", dimension: " + preferredDimension);
183             return id;
184         } else {
185             LOG.debug("About to look for translations.  Input asset id: " + id + ", dimension set: " + dimensionSetName
186                     + ", preferred dimension: " + preferredDimension);
187         }
188 
189         // *****************************************************************************
190         // The core business logic of this helper class is encapsulated in these
191         // 3 lines
192         final DimensionSetInstance dimset = getDimensionSet(dimensionSetName);
193 
194         return findTranslation(id, preferredDimension, dimset);
195     }

208 
209         final DimensionFilterInstance filter = _getPopulatedDimensionFilter(dimset, preferredDimension);
210         AssetId translated = findTranslation(id, filter);
211         if (translated == null)

213                     + preferredDimension + ".");
214         return translated;
215     }

220     private boolean _isInputAssetDimensionPreferred(final AssetId id, final long preferredDimension) {
221         final Dimension dim = getLocaleForAsset(id);
222         if (dim == null) {
223             return true; // if locale not found, tell that the asset is expected
224         }
225         // locale
226         return dim.getId().getId() == preferredDimension;
227     }

229     private static final PreparedStmt FIND_DIMSET_FOR_SITE_PREPAREDSTMT = new PreparedStmt(
230             "select ds.id as id from DimensionSet ds, Publication p, AssetPublication ap where p.name = ? and p.id = ap.pubid and ap.assetid = ds.id and ds.status != 'VO' order by ds.updateddate",
231             Arrays.asList("DimensionSet", "AssetPublication", "Publication"));
232 
233     static {
234         FIND_DIMSET_FOR_SITE_PREPAREDSTMT.setElement(0, "Publication", "name");
235     }

237     /**
238      * Locates a single dimension set in a site. If no match is found, an
239      * exception is thrown. If more than one match is found, an exception is
240      * thrown.
241      * 
242      * @param site site containing a dimension set
243      * @return DimensionSet ID
244      */
245     public long locateDimensionSetForSite(final String site) {
246         if (StringUtils.isBlank(site)) {
247             throw new IllegalArgumentException("Required site name missing");
248         }
249         final StatementParam params = FIND_DIMSET_FOR_SITE_PREPAREDSTMT.newParam();
250         params.setString(0, site);
251         final IListIterable list = SqlHelper.select(ics, FIND_DIMSET_FOR_SITE_PREPAREDSTMT, params);
252 
253         final int numRows = list.size();
254         if (numRows == 0) {
255             throw new IllegalStateException(
256                     "A DimensionSet has not been defined for site '"
257                             + site
258                             + "'. Cannot determine any translation unless some locales (Dimensions) are enabled for that site. Aborting operation.");
259         }
260         if (numRows > 1) {
261             final StringBuilder msg = new StringBuilder("More than one dimension set found in site " + site
262                     + ".  Exactly one is expected.  Dimension set ids: ");
263             for (final Row row : list) {
264                 final String id = row.getString("id");
265                 LogDep.logDep(ics, "DimensionSet", id);
266                 msg.append(id).append(" ");
267             }
268             throw new IllegalStateException(msg.append(".").toString());
269         }
270 
271         final String id = list.iterator().next().getString("id");
272         LogDep.logDep(ics, "DimensionSet", id);
273         return Long.parseLong(id);
274     }

276     private DimensionFilterInstance _getPopulatedDimensionFilter(final DimensionSetInstance dimset,
277             final long dimensionId) {
278 
279         // Set the filter's preferred dimension
280         // Equivalent to:
281         // %><dimensionset:asset assettype="Dimension"
282         // assetid="<%=localeDimensionId%>" /><%
283         final Dimension preferredDimension = getDM().loadDimension(dimensionId);
284         if (preferredDimension == null) {
285             throw new RuntimeException("Attempted to load Dimension with id " + dimensionId + " but it came back null");
286         }
287         return _getPopulatedDimensionFilter(dimset, preferredDimension);
288     }

290     private DimensionFilterInstance _getPopulatedDimensionFilter(final DimensionSetInstance dimset,
291             final Dimension preferredDimension) {
292         DimensionFilterInstance filter;
293         try {
294             filter = dimset.getFilter();
295         } catch (final DimensionException e) {
296             throw new RuntimeException("Could not get Dimension Filter from DimensionSet", e);
297         }
298 
299         filter.setDimensonPreference(Collections.singletonList(preferredDimension));
300         return filter;
301     }

303     private DimensionSetInstance _getDimensionSet(final long dimSetId) {
304         final String DIMSET_OBJ_NAME = "LocaleUtils:findTranslation:theDimensionSet:DimensionSet";
305         // Load the site-specific DimensionSet asset
306         ics.SetObj(DIMSET_OBJ_NAME, null); // clear first
307         final FTValList args = new FTValList();
308         args.put("NAME", DIMSET_OBJ_NAME);
309         args.put("TYPE", "DimensionSet");
310         args.put("OBJECTID", Long.toString(dimSetId));
311         args.put("EDITABLE", "FALSE");
312         ics.runTag("ASSET.LOAD", args);
313 
314         if (ics.GetErrno() < 0) {
315             throw new IllegalStateException("Could not load dimension set.  Errno: " + ics.GetErrno());
316         }
317 
318         final Object o = ics.GetObj(DIMSET_OBJ_NAME);
319         ics.SetObj(DIMSET_OBJ_NAME, null);
320         if (o == null) {
321             throw new IllegalStateException("Could not load dimension set but we got no errno... unexpected...");
322         }
323 
324         DimensionSetInstance dimset;
325         if (o instanceof DimensionSetInstance) {
326             dimset = (DimensionSetInstance) o;
327         } else {
328             throw new IllegalStateException("Loaded an asset that is not a DimensionSetInstance.");
329         }
330         return dimset;
331     }

486     @Override
487     public AssetId findTranslation(AssetId id, DimensionFilterInstance filter) {
488         // Get the relatives using the appropriate filter
489         final Collection<AssetId> relatives = getDAM().getRelatives(id, filter, "Locale");
490         // *****************************************************************************
491 
492         // make the result pretty
493         if (relatives == null) {
494             LOG.debug("No translation found for asset " + id + ".");
495             return null;
496         } else {
497             switch (relatives.size()) {
498                 case 0: {
499                     LOG.debug("No translation found for " + id + ".");
500                     // Note May 4, 2010 by Tony Field - this had been changed to
501                     // return the input ID but that
502                     // is incorrect. The contract clearly states that null is to
503                     // be returned if no matching
504                     // relatives are found. When null is returned and it is not
505                     // expected, often the incorrect
506                     // dimension filter is configured.
507                     return null;
508                 }
509                 case 1: {
510                     final AssetId relative = relatives.iterator().next();
511                     LOG.trace("LocaleUtils.findTranslation: RELATIVE FOUND... " + relative.getType() + " '"
512                             + relative.getId() + "' // errno = " + ics.GetErrno());
513                     return relative;
514 
515                 }
516                 default: {
517                     throw new IllegalStateException("found more than one translation for asset " + id
518                             + " and that is not supposed to be possible.");
519                 }
520             }
521         }
522     }

524     /**
525      * Return a dimension filter instance corresponding to the dimension set
526      * specified by the user (or discovered by the tag). The dimension filter is
527      * configured with the preferred dimensions of the user (also configured).
528      * <p>
529      * The preferred locales are identified by checking the following locations,
530      * in the order specified:
531      * <ol>
532      * <li>set by the lang attribute by locale name
533      * <li>detected by finding the locale dimension id in the ics variable
534      * "locale"
535      * <li>detected by finding the locale name in the ics variable "locale"
536      * <li>detected by finding the locale dimension id in the ics session
537      * variable "locale"
538      * <li>detected by finding the locale name in the ics session variable
539      * "locale"
540      * <li>detected by reading the Accept-Language header
541      * </ol>
542      * The dimension set is identified by checking in the following places, in
543      * order:
544      * <ol>
545      * <li>set by the dimset attribute by name of dimension set
546      * <li>set by dimset attribute by the id of the dimension set
547      * <li>looked up by finding the site name in the ics variable "site" and
548      * loading the single dimension set associated with that site
549      * </ol>
550      * 
551      * @return a dimension filter, configured with the set preferred locales, or
552      *         null, if either the dimension set or the preferred dimensions
553      *         could not be found (with extensive errors)
554      */
555     public DimensionFilterInstance getDimensionFilter(String site) {
556 
557         DimensionFilterInstance filter;
558         try {
559             DimensionSetInstance dimSet = locateDimensionSetInstanceForSite(site);
560             if (dimSet == null) {
561                 if (LOG.isTraceEnabled()) {
562                     LOG.trace("no DimensionSet returned from getDimensionSet().");
563                 }
564                 return null;
565             }
566             Collection<AssetId> preferredLocales = getPreferredLocales();
567 
568             filter = DimensionUtils.getDimensionFilter(DimensionUtils.getDM(ics), preferredLocales, dimSet);
569             if (LOG.isDebugEnabled()) {
570                 LOG.debug("Located dimension filter: " + filter + " in dimensionSet " + dimSet
571                         + " with preferred locales: " + preferredLocales + " ");
572             }
573         } catch (DimensionException e) {
574             LOG.error("Could not locate dimension filter", e);
575             filter = null;
576         } catch (RuntimeException e) {
577             LOG.error("Could not locate dimension filter", e);
578             filter = null;
579         }
580         return filter;
581     }

583     /**
584      * Get the locale that the user explicitly specified. If not set, null is
585      * returned.
586      * 
587      * @return the id of the locale that the user explicitly set. Handles
588      *         setting by name or assetid.
589      */
590     protected final AssetId getExplicitlySpecifiedLocale() {
591 
592         String localeName = ics.GetVar(langVar);
593         // check for explicitly specified by name
594         if (StringUtils.isNotBlank(localeName)) {
595             Dimension d = DimensionUtils.getDimensionForName(ics, localeName);
596             if (d != null) {
597                 LOG.trace("Preferred locale explicitly set to " + localeName);
598                 return d.getId();
599             }
600         }
601         return null;
602     }

604     /**
605      * Get the ordered list of preferred locales that the user wants. Multiple
606      * attempts are made to figure out the right locale.
607      * 
608      * @return collection of asset identifiers of the preferred locales
609      */
610     protected final Collection<AssetId> getPreferredLocales() {
611         AssetId result = getExplicitlySpecifiedLocale();
612         if (result != null)
613             return Collections.singleton(result);
614 
615         // next, check for implicitly specified by ID using locale variable
616         String l = ics.GetVar(localeVar);
617         try {
618             long localeIdFromVar = Long.parseLong(l);
619             DimensionManager dm = DimensionUtils.getDM(ics);
620             Dimension d = dm.loadDimension(localeIdFromVar);
621             if (d != null) {
622                 LOG.trace("Preferred locale detected in ICS context using 'locale' variable: " + localeIdFromVar);
623                 return Collections.singletonList(d.getId());
624             }
625         } catch (NumberFormatException e) {
626             // maybe it's a locale name...
627             try {
628                 Dimension d = DimensionUtils.getDimensionForName(ics, localeVar);
629                 if (d != null) {
630                     LOG.trace("Preferred locale detected in ICS context using 'locale' variable: " + localeVar);
631                     return Collections.singletonList(d.getId());
632                 }
633             } catch (Exception ex) {
634                 // nope... don't worry, we'll find it....
635             }
636         }
637 
638         // next, check for implicitly specified by ID using locale session
639         // variable
640         String localeSSVar = ics.GetSSVar(localeVar);
641         try {
642             long localeIdFromSSVar = Long.parseLong(localeSSVar);
643             DimensionManager dm = DimensionUtils.getDM(ics);
644             Dimension d = dm.loadDimension(localeIdFromSSVar);
645             if (d != null) {
646                 LOG.trace("Preferred locale detected in ICS context using 'locale' session variable: "
647                         + localeIdFromSSVar);
648                 return Collections.singletonList(d.getId());
649             }
650         } catch (NumberFormatException e) {
651             // maybe it's a locale name...
652             try {
653                 Dimension d = DimensionUtils.getDimensionForName(ics, localeSSVar);
654                 if (d != null) {
655                     LOG.trace("Preferred locale detected in ICS context using 'locale' session variable: "
656                             + localeSSVar);
657                     return Collections.singletonList(d.getId());
658                 }
659             } catch (Exception ex) {
660                 // nope... don't worry, we'll find it....
661             }
662         }
663 
664         // finally, get the locale from the servlet request's Accept-Language
665         // header..
666         List<AssetId> preferredLocales = new ArrayList<AssetId>();
667         @SuppressWarnings({ "rawtypes", "deprecation" })
668         Enumeration locales = ics.getIServlet().getServletRequest().getLocales();
669         while (locales.hasMoreElements()) {
670             Locale locale = (Locale) locales.nextElement();
671             if (locale != null) {
672                 String localeName = locale.toString();
673                 if (localeName != null && localeName.length() > 0) {
674                     try {
675                         Dimension dimension = DimensionUtils.getDimensionForName(ics, localeName);
676                         if (dimension != null) {
677                             preferredLocales.add(dimension.getId());
678                             LOG.trace("Found registered locale in user's Accept-Language header (or default): "
679                                     + localeName);
680                         }
681                     } catch (RuntimeException e) {
682                         // don't care if the dimension is not in the system -
683                         // they probably won't all be there
684                         // and we're guessing anyway, so it's okay.
685                         LOG.trace(
686                                 "Found a locale in the user's Accept-Language header, but it was not registered as a dimension: "
687                                         + localeName + " (this is not usually an error)", e);
688                     }
689                 }
690             }
691         }
692         return preferredLocales;
693     }
694 
695     /**
696      * Returns the DimensionSetInstance. This implementation expects one
697      * DimensionSet enabled for the current site.
698      * 
699      * @param site current site
700      * @return dimension set instance
701      */
702     public final DimensionSetInstance locateDimensionSetInstanceForSite(String site) {
703         try {
704             if (StringUtils.isNotBlank(site)) {
705                 long discoveredId = locateDimensionSetForSite(site);
706                 LOG.trace("Auto-discovered dimension set because there is only one in site " + site + ": DimensionSet:"
707                         + discoveredId);
708                 return _getDimensionSet(discoveredId);
709             }
710         } catch (RuntimeException e) {
711             if (LOG.isTraceEnabled())
712                 LOG.trace("Could not auto-discover dimensionset: " + e);
713         }
714         return null;
715     }
