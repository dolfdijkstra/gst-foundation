42  /**
43   * Simple navigation service implementation that loads objects from the Site Plan.
44   * 
45   * Reads the full site plan tree in one query. It does not filter assets by site.
46   * 
47   * Nodes are instantiated via a dedicated method where you can load any data that
48   * is required; you get to (define and) use your own AssetNode implementation.
49   * 
50   * This NavService implementation REQUIRES that the SitePlan models your navigational
51   * structures after the following design:
52   * 
53   * SitePlan root node (SiteNavigation asset)
54   *        |
55   *        ----------- Nav Structure "A" Placeholder (Page asset)
56   *        |                     |
57   *        |                     |----------- Webpage A.1 (Page asset)
58   *        |                     |                  |
59   *        |                     |                  |----------- Webpage A.1.1 (Page asset)
60   *        |                     |                                    |----------- (...) (Page assets)
61   *        |                     |
62   *        |                     |----------- Webpage A.2 (Page asset)
63   *        |                     |                  |
64   *        |                     |                  |----------- (...) (Page assets)
65   *        |                     |
66   *        |                     |----------- Webpage A.3 (Page asset)
67   *        |                     |                  |
68   *        |                     |                  |----------- (...) (Page assets)
69   *        |
70   *        ----------- Nav Structure "B" Placeholder (Page asset)
71   *        |                     |
72   *        |                     |----------- Webpage B.1 (Page asset)
73   *        |                     |                  |
74   *        |                     |                  |----------- (...) (Page assets)
75   *        |                     |                  
76   *        |                     |----------- Webpage B.2 (Page asset)
77   *        |                     |                  |
78   *        |                     |                  |----------- (...) (Page assets)
79   *        |                     |                                    
80   *        |                     |----------- Webpage A.3 (Page asset)
81   *        |                                        |
82   *        |                                        |----------- (...) (Page assets)
83   *        |
84   *        ----------- (...) (Other nav structures)
85   *
86   * ... where:
87   * 
88   * - SiteNavigation nodes will always be excluded from a Webpage node's definitive breadcrumb.
89   * - SiteNavigation nodes are not part of any nav structure.
90   * - Nav Structure Placeholder nodes will always be excluded from a Webpage node's definitive breadcrumb.
91   * - Nav Structure Placeholder are not part of a nav structure, they are just the entry point to it.
92   *
93   *
94   *
95   *
96   * 
97   * @author Tony Field
98   * @since 2016-07-06
99   */
100 public abstract class SitePlanNavService<N extends AssetNode<N> & ConfigurableNode<N>> implements NavService<N, AssetId, AssetId> {

102 	private static final Logger LOG = LoggerFactory.getLogger(SitePlanNavService.class);

104 	private boolean _initialized;

106     private final ICS ics;
107     private final TemplateAssetAccess dao;
108     private final String sitename;
109 	private final Long pubId;
110     private final Map<AssetId, List<N>> nodesById = new HashMap<>();
111 
112     // Sure, we could join with PUBLICATION on the NAVIGATION_TREE_DUMP, but
113     // this way we leverage cache further.
114     private final static PreparedStmt FIND_PUBID = new PreparedStmt(
115             "select id from Publication where name = ?",
116             Arrays.asList("Publication"));
117 
118     private final static PreparedStmt NAVIGATION_TREE_DUMP = new PreparedStmt(
119             "select spt.* from SITEPLANTREE spt, ASSETPUBLICATION ap " +
120             "where " +
121             "spt.ncode = 'Placed' and " +
122             "spt.otype = ap.assettype and " +
123             "spt.oid = ap.assetid and " +
124             "((ap.pubid = ?) OR (ap.pubid = 0)) " + // Pages cannot be shared across sites in 12c, but let's play it safe, just in case that changes
125             "order by spt.nparentid, spt.nrank",
126             Arrays.asList("page", "siteplantree", "assetpublication", "Page", "SitePlanTree", "AssetPublication", "PAGE", "SITEPLANTREE", "ASSETPUBLICATION"));
127     
128     static {
129     	FIND_PUBID.setElement(0, "Publication", "name");
130     	NAVIGATION_TREE_DUMP.setElement(0, "ASSETPUBLICATION", "pubid");
131     }
132     
133     private final Long _getPubId(String sitename) {
134         final StatementParam params = FIND_PUBID.newParam();
135         params.setString(0, sitename);
136 
137         LOG.debug("Executing query to find out pubid for site name {}", sitename);
138                 
139         IListIterable results = SqlHelper.select(this.ics, FIND_PUBID, params);
140         if (results.size() == 0) {
141         	LOG.warn("There is no Publication whose name is '{}'", sitename);
142         	return null;
143         } else if (results.size() > 1) {
144         	throw new IllegalStateException("There cannot be 2+ sites (publications) in WCS with the same name, yet there seem to be " + results.size() + " sites whose name is '" + sitename + "'");
145         } else {
146         	Row row = results.iterator().next();
147         	return row.getLong("id");
148         }
149     }

169     	this(ics, dao, null);

172     public SitePlanNavService(ICS ics, TemplateAssetAccess dao, String theSitename) {
173         this.ics = ics;
174         this.dao = dao;
175 
176         if (!Utilities.goodString(theSitename)) {
177     		// Fallback to ICS variable "site", if such
178         	theSitename = ics.GetVar("site");
179         	if (!Utilities.goodString(theSitename)) {
180         		throw new IllegalStateException("Missing argument sitename. Nav structure cannot be built unless you specify the name of the site (publication) it is for.");
181         	}
182     	}
183 		Long aPubId = _getPubId(theSitename);
184 		if (aPubId == null) {
185 			throw new IllegalStateException("Cannot determine pubid for site '" + theSitename + "'. Nav structure cannot be built unless you specify the (valid) name of the site (publication) it is for.");
186 		}
187         
188 		this.pubId = aPubId;
189 		this.sitename = theSitename;
190     }
191 
192     private synchronized void _initializeNavStructure() {
193     	if (!this._initialized) {
194 	        StatementParam params = NAVIGATION_TREE_DUMP.newParam();
195 	        params.setLong(0, pubId);
196 	
197 	        // read the site plan tree in one massive query
198 	        Map<Long, SitePlanTreeData> rowMap = new HashMap<>();
199 	        Map<Long, List<SitePlanTreeData>> childrenMap = new HashMap<>();
200 	
201 	        LOG.debug("Executing SitePlan query for gathering data for nav service...");
202 	                
203 	        for (Row row : SqlHelper.select(ics, NAVIGATION_TREE_DUMP, params)) {
204 	            SitePlanTreeData nodeInfo = new SitePlanTreeData(row);
205 	            LOG.debug("Processing SitePlan row: {}", nodeInfo);
206 	            rowMap.put(nodeInfo.nid, nodeInfo);
207 	            LOG.debug("Added row {} to SitePlan rows map under key {}", nodeInfo, nodeInfo.nid);
208 	            List<SitePlanTreeData> children = childrenMap.get(nodeInfo.nparentid);
209 	            if (children == null) {
210 	            	// Initialize the list of children for the current row's parent
211 	            	children = new ArrayList<SitePlanTreeData>();
212 	            	childrenMap.put(nodeInfo.nparentid, children);
213 	            }
214 	            // Add the current row to its parent's list of children  
215 	            children.add(nodeInfo);
216 	            LOG.debug("Added SPT row {} to the list of children of SPT (parent) row {}. That list now looks like this: {}", nodeInfo, nodeInfo.nparentid, children);
217 	        }
218 	
219 	        // create Node objects
220 	        Map<Long, N> nidNodeMap = new HashMap<Long, N>();
221 	        for (long nid : rowMap.keySet()) {
222 	        	LOG.debug("Will invoke createAssetNode for asset id {}", rowMap.get(nid).assetId);
223 	        	N node = createAssetNode(rowMap.get(nid).assetId);
224 	        	LOG.debug("AssetNode created for asset {}: {}", rowMap.get(nid).assetId, node);
225 	            
226 	            // Log a dependency with every node (asset) we populate
227 	            LogDep.logDep(ics, node.getId());
228 	        	LOG.debug("Logged dependency for asset {} inside nav service...", rowMap.get(nid).assetId);
229 	            
230 	            nidNodeMap.put(nid, node);
231 	            LOG.debug("Added node {} to nodes map under key {}", node, nid);
232 	            
233 	            // Stash for later. Probably won't have many duplicates so optimize
234 	            AssetId assetId = node.getId();
235 	            List<N> a1 = nodesById.get(assetId);
236 	            if (a1 == null) {
237 	            	a1 = Stream.of(node).collect(Collectors.toList());
238 	                nodesById.put(assetId, a1);
239 	            } else {
240 	            	a1.add(node);
241 	            }
242 	            
243 	            LOG.debug("nodesById map: {}", nodesById);
244 	
245 	        }
246 	
247 	        // hook up parent-child relationships, starting from the list
248 	        // of nodes with a parent
249 	        for (long nparentid : childrenMap.keySet()) {
250 	        	LOG.debug("Processing parent-child relationships for SitePlanTree row with nid = {}", nparentid);
251 	        	N parent = nidNodeMap.get(nparentid);
252 	        	if (parent != null) {
253 	        		LOG.debug("AssetNode for nid {} is: {}", nparentid, parent);
254 	        		List<SitePlanTreeData> children = childrenMap.get(nparentid);
255 	        		if (children != null) {
256 	        			LOG.debug("List of children for SPT whose nid = {} is: {}", nparentid, children);
257 	        			for (SitePlanTreeData childRow : children) {
258 		        			N child = nidNodeMap.get(childRow.nid);
259 		        			if (child != null) {
260 		        				parent.addChild(child);
261 		        				child.setParent(parent);
262 		        				LOG.debug("Bound together parent node {} and child node {} as per SPT entry {}", parent, child, childRow);  
263 		        			} else {
264 		        				LOG.warn("There could be a problem here... we registered SPT row {} as a child of parent row {} but we did not instantiate a node for that child?", childRow, nparentid);
265 		        			}
266 		        		}
267 	        		} else {
268 	        			LOG.warn("Not sure how we ended up with a a parent node in the children nodes map with no children whatsoever.");
269 	        		}
270 	        	} else {
271 	        		LOG.warn("We have a list of children for SPT entry whose nid = {}. However, there is not a node matching that SPT entry's asset ({}). This is a bit weird, but also legitimate (for instance, the query excludes SiteNavigation assets)", nparentid, rowMap.get(nparentid));
272 	        	}
273 	        }
274 	        
275 	        this._initialized = true;
276     	} else {
277     		LOG.debug("Nav Structure had been already initialized for NavService instance {}", this);
278     	}
279     }

281     private static class SitePlanTreeData {
282         final long nid;
283         final long nparentid;
284         final int nrank;
285         final AssetId assetId;
286 
287         SitePlanTreeData(Row row) {
288             nid = row.getLong("nid");
289             nparentid = row.getLong("nparentid");
290             nrank = row.getInt("nrank");
291             assetId = AssetIdUtils.createAssetId(row.getString("otype"), row.getLong("oid"));
292         }
293 
294         @Override
295         public String toString() {
296             return "SitePlanTreeData{" +
297                     "nid=" + nid +
298                     ", nparentid=" + nparentid +
299                     ", nrank=" + nrank +
300                     ", assetId=" + assetId +
301                     '}';
302         }
303     }

305     public List<N> getNav(AssetId sitePlan) {
306         if (sitePlan == null) {
307             throw new IllegalArgumentException("Null param not allowed");
308         }
309 
310         // Initialize the nav structure
311         _initializeNavStructure();
312 
313         // find the requested structure
314         List<N> spNodes = nodesById.get(sitePlan);
315         if (spNodes == null) throw new IllegalArgumentException("Could not locate nav structure corresponding to "+sitePlan);
316         if (spNodes.size() > 1) throw new IllegalStateException("Cannot have more than one site plan node with the same id in the tree");
317         N requestedRoot = spNodes.get(0); // never null
318 
319         // return the loaded children of the structure root
320         return (List<N>) requestedRoot.getChildren();
321     }

323     public List<N> getBreadcrumb(AssetId id) {
324         if (id == null) {
325             throw new IllegalArgumentException("Cannot calculate breadcrumb of a null asset");
326         }
327         
328         // Initialize the nav structure
329         _initializeNavStructure();
330 
331         Collection<BreadcrumbCandidate<N>> breadcrumbs = new ArrayList<>();
332         LOG.debug("Obtaining all nodes whose ID matches {}", id);
333         List<N> nodes = nodesById.get(id);
334         if (nodes != null) {
335         	// Build all possible breadcrumbs
336 	        for (N node : nodesById.get(id)) {
337 	            breadcrumbs.add(new BreadcrumbCandidate<N>(getBreadcrumbForNode(node)));
338 	        }
339 	        
340 	        // Choose the preferred breadcrumb. Users may override this method
341 	        // to implement their own choosing logic.
342 	        List<N> breadcrumb = chooseBreadcrumb(breadcrumbs);
343 	        LOG.debug("This is the preferred breadcrumb for ID {}: {}", id, breadcrumb);
344 	        	        
345 	        return breadcrumb;
346         } else {
347         	LOG.info("Didn't find any node in this nav structure whose ID matched {}, thus a breadcrumb cannot be calculated. These are all available nodes: {}", id, this.nodesById);
348         	LOG.debug("Node not found, getBreadcrumb will return null");
349         	return null;
350         }
351 
352         
353     }

355     /**
356      * Get the breadcrumb corresponding to the specified node.
357      *
358      * Default implementation simply uses the specified node's parents.
359      *
360      * @param node the node whose breadcrumb needs to be calculated
361      * @return the breadcrumb
362      */
363     protected List<N> getBreadcrumbForNode(N node) {
364         List<N> ancestors = new ArrayList<>();
365         do {
366             ancestors.add(node);
367             node = node.getParent();
368         } while (node != null);
369         Collections.reverse(ancestors);
370         return ancestors;
371     }

373     /**
374      * Determines a node's "preferred" breadcrumb from a list of "candidates".
375      *
376      * This default implementation simply returns the first one returned by the specified
377      * collection's iterator.
378      * 
379      * However, you may override this method in order to implement your own ad-hoc logic for
380      * determining the preferred breadcrumb.
381      * 
382      * The candidates passed into this method have already parsed the raw breadcrumb so to split it
383      * up in the due SiteNavigation node, Nav Structure Placeholder node and the clean breadcrumb path.
384      * 
385      * @param candidates The candidates you get to choose the preferred breadcrumb from.
386      * @return The preferred breadcrumb (list of nodes).
387      */
388     protected List<N> chooseBreadcrumb(Collection<BreadcrumbCandidate<N>> candidates) {
389         return candidates.iterator().next().getBreadcrumb();
390     }
391     
392     protected static class BreadcrumbCandidate<N extends AssetNode<N>> {
393     	private List<N> breadcrumb;
394     	private N siteNavigation;
395     	private N navStructurePlaceholder;
396     	
397     	public BreadcrumbCandidate(List<N> rawBreadcrumb) {
398     		if (rawBreadcrumb.size() < 2) {
399     			throw new IllegalArgumentException("Breadcrumb candidate is not valid, insufficient nodes.");
400     		}
401     		this.siteNavigation = rawBreadcrumb.get(0);
402     		if (!this.siteNavigation.getId().getType().equals("SiteNavigation")) {
403     			throw new IllegalArgumentException("Breadcrumb candidate is not valid, first node must be the SiteNavigation node, got this instead: " + this.siteNavigation);
404     		}
405     		this.navStructurePlaceholder = rawBreadcrumb.get(1);
406     		if (!this.navStructurePlaceholder.getId().getType().equals("Page")) {
407     			throw new IllegalArgumentException("Breadcrumb candidate is not valid, second node must be the Nav Structure Placeholder (Page) node, got this instead: " + this.navStructurePlaceholder);
408     		}
409     		this.breadcrumb = rawBreadcrumb.subList(2, rawBreadcrumb.size());
410     	}
411     	
412     	public N getNavStructurePlaceholder() {
413     		return this.navStructurePlaceholder;
414     	}
415     	
416     	public List<N> getBreadcrumb() {
417     		return this.breadcrumb;
418     	}
419     }
