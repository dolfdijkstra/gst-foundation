41  /**
42   * ServletContextListener that loads and configures the AppContext for this
43   * application.
44   *
45   * It is NOT configured to auto-load into the servlet context by way of the
46   * {@code @WebListener} annotation.  Overriding of this class is allowed
47   * but not advised as it is now deprecated.
48   * 
49   * GSF's own Groovy Loader / support has been deprecated, must not be used anymore.
50   * 
51   * @author Dolf Dijkstra
52   * @author Tony Field
53   * 
54   * @deprecated Replaced by the brand new ServletContextLoader (which requires Servlet 3.0 and differs in usage)
55   * 
56   */
57  public class WebAppContextLoader implements ServletContextListener {
58      public static final String CONTEXTS = "gsf-contexts";
59  
60      protected static final Logger LOG = LoggerFactory.getLogger("tools.gsf.legacy.controller.support.WebAppContextLoader");
61      private static final Class<?>[] ARGS = new Class[] { ServletContext.class, AppContext.class };
62  
63      public void contextInitialized(final ServletContextEvent sce) {
64          final ServletContext context = sce.getServletContext();
65          if (context.getMajorVersion() == 3 && context.getMinorVersion() < 0) {
66              throw new IllegalStateException("Servlet Container is configured for version less than 3.0. This ServletContextListener does not support 2.x and earlier as the load order of Listeners is not guaranteed.");
67          }
68          configureWebAppContext(context);
69      }
70  
71      private AppContext configureWebAppContext(final ServletContext context) {
72  
73          AppContext parent;
74  
75          final ClassLoader cl = Thread.currentThread().getContextClassLoader();
76  
77          // start with init param
78          parent = configureFromInitParam(context, cl);
79          if (parent != null) {
80              LOG.info("AppContext initialized from init param. Class: {}", parent.getClass().getName());
81          }
82  
83          // try locator
84          if (parent == null) {
85              parent = configureFromServiceLocator(context, cl);
86              if (parent != null) {
87                  LOG.info("AppContext initialized from service locator. Class: {}", parent.getClass().getName());
88              }
89          }
90  
91          // use default
92          if (parent == null) {
93              parent = new DefaultWebAppContext(context);
94              parent.init();
95              LOG.info("AppContext initialized using default class: {}", parent.getClass().getName());
96          }
97  
98          // register it
99          context.setAttribute(WebAppContext.WEB_CONTEXT_NAME, parent);
100 
101         return parent;
102     }
103 
104     private static final String PREFIX = "META-INF/";
105 
106     private AppContext configureFromServiceLocator(ServletContext context, ClassLoader cl) {
107         String fullName = PREFIX + CONTEXTS;
108 
109         List<String> init = new LinkedList<>();
110         Enumeration<URL> configs;
111         try {
112             configs = cl.getResources(fullName);
113         } catch (IOException e) {
114             LOG.warn("Exception when loading the service descriptor for the AppContext from the classpath.", e);
115             return null;
116         }
117 
118         if (configs.hasMoreElements()) {
119             URL u = configs.nextElement();
120             try (InputStream in = u.openStream();
121                  BufferedReader r = new BufferedReader(new InputStreamReader(in, "utf-8"))) {
122                 String s;
123                 while ((s = r.readLine()) != null) {
124                     if (StringUtils.isNotBlank(s) && !StringUtils.startsWith(s, "#")) {
125                         init.add(s);
126                     }
127                 }
128             } catch (IOException e) {
129                 throw new RuntimeException("Error reading configuration file", e);
130             }
131             return this.createFromString(context, cl, init.toArray(new String[init.size()]));
132         }
133         return null;
134     }
135 
136     /**
137      * Creates a AppContext based on the init parameter {@link #CONTEXTS}.
138      * 
139      * @param context servlet context
140      * @param cl class loader
141      * @return the AppContext as configured from the web app init parameter.
142      */
143     protected AppContext configureFromInitParam(final ServletContext context, final ClassLoader cl) {
144         final String init = context.getInitParameter(CONTEXTS);
145         AppContext parent = null;
146 
147         if (init != null) {
148             parent = createFromString(context, cl, init.split(","));
149         }
150         return parent;
151     }
152 
153     private AppContext createFromString(final ServletContext context, final ClassLoader cl, final String[] c) {
154         AppContext parent = null;
155 
156         if (c != null) {
157             for (int i = c.length - 1; i >= 0; i--) {
158                 try {
159                     final AppContext n = createAppContext(cl, c[i], context, parent);
160                     if (n != null) {
161                         parent = n;
162                     }
163                 } catch (IllegalArgumentException
164                         | InstantiationException
165                         | IllegalAccessException
166                         | InvocationTargetException
167                         | SecurityException
168                         | NoSuchMethodException
169                         | ClassNotFoundException e) {
170                     LOG.warn("Couldn't create the app context", e);
171                 }
172             }
173         }
174         return parent;
175     }
176 
177     /**
178      * @param cl the classloader to load the class.
179      * @param c the class name, class needs to implement a constructor with a
180      *            <tt>context</tt> and a <tt>parent</tt> {@link #ARGS}.
181      * @param context the web context
182      * @param parent parent AppContext, null is none
183      * @return the created AppContext
184      * @throws ClassNotFoundException class not found
185      * @throws SecurityException security exception
186      * @throws NoSuchMethodException no method exists
187      * @throws InstantiationException could not instantiate
188      * @throws IllegalAccessException no access to perform this operation
189      * @throws InvocationTargetException exception from invocation
190      */
191     AppContext createAppContext(final ClassLoader cl, final String c, final ServletContext context,
192             final AppContext parent) throws ClassNotFoundException, SecurityException, NoSuchMethodException,
193             InstantiationException, IllegalAccessException, InvocationTargetException {
194         @SuppressWarnings("unchecked")
195         final Class<AppContext> cls = (Class<AppContext>) cl.loadClass(c);
196         final Constructor<AppContext> ctr = cls.getConstructor(ARGS);
197         AppContext n;
198         n = ctr.newInstance(context, parent);
199         n.init();
200         return n;
201     }
202 
203     public void contextDestroyed(final ServletContextEvent sce) {
204         sce.getServletContext().removeAttribute(WebAppContext.WEB_CONTEXT_NAME);
205         LOG.info("AppContext un-registered from servlet context.");
206     }
