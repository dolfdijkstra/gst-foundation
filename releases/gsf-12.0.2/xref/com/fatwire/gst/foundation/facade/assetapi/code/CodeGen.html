70          public void printAssetTypeDef(AssetTypeDef def) {
71              if (def == null) {
72                  return;
73              }
74              if (def.getSubtype() == null && def.getProperties().getIsFlexAsset()) {
75                  return;
76              }
77              String s = def.getSubtype() == null ? "" : ("/" + def.getSubtype());
78  
79              out.print(def.getName() + s);
80              out.print("   // description: '" + def.getDescription() + "'");
81              out.print(" plural: '" + def.getPlural() + "'");
82              out.print(" canbechild: " + def.getCanBeChild() + "");
83              AssetTypeDefProperties p = def.getProperties();
84  
85              out.print(" canAddSubtypes: " + p.getCanAddSubtypes());
86              String type = p.getIsAssetmakerAsset() ? "AssetmakerAsset" : "Unknown Type";
87              if (p.getIsCoreAsset()) {
88                  type = "CoreAsset";
89              }
90              if (p.getIsFlexAsset()) {
91                  type = "FlexAsset";
92              }
93              out.print(" type: '" + type + "'");
94              out.println(" nameMustUnique: " + p.getIsNameMustUnique());
95  
96              printAttributes(def.getAttributeDefs());
97              printParentDefs(def.getParentDefs());
98              printAssocations(def.getAssociations());
99  
100             out.println();
101             out.println();
102         }

230         protected String toVarName(String name) {
231             if (name == null || name.length() == 0) {
232                 return "noname";
233             }
234             if (KEYWORDS.contains(name)) {
235                 return name + "_";
236             }
237             char[] t = name.replace("-", "_").toCharArray();
238             t[0] = Character.toLowerCase(t[0]);
239             return new String(t);
240         }

314 
315             switch (t) {
316                 case INT:
317                     return "Integer";
318                 case FLOAT:
319                     return "Double";
320                 case STRING:
321                     return "String";
322                 case LONG:
323                     return "Long";
324                 case DATE:
325                     return "Date";
326                 case MONEY:
327                     return "Double";
328                 case LARGE_TEXT:
329                     return "String";
330                 case ASSET:
331                     return "AssetId";
332                 case ASSETREFERENCE:
333                     return "AssetId";
334                 case BLOB:
335                     return "BlobObject";
336                 case URL:
337                     return "BlobObject";
338                 case ARRAY:
339                     return "List<?>";
340                 case STRUCT:
341                     return "Map<?,?>";
342                 case LIST:
343                     return "List<?>";
344                 case ONEOF:
345                     return "Object";

549     void gen(ICS ics, JspWriter out) throws AssetAccessException, IOException {
550 
551         Session s = SessionFactory.getSession(ics);
552         AssetTypeDefManager mgr = (AssetTypeDefManager) s.getManager(com.fatwire.assetapi.def.AssetTypeDefManager.class
553                 .getName());
554 
555         String ats = ics.GetVar("assettype");
556         Generator generator = null;
557         String what = ics.GetVar("what");
558         if ("assetapicode".equals(what)) {
559             generator = new AssetApiGen(out);
560         } else if ("def".equals(what)) {
561             generator = new DefPrint(out);
562         } else if ("gsfjspcode".equals(what)) {
563             generator = new GsfJspGen(out);
564         } else if ("gsfjavacode".equals(what)) {
565             generator = new GsfJavaGen(out);
566         }
567 
568         if (ats != null && generator != null) {
569             for (String at : ats.split(";")) {
570                 String[] p = at.split(":");
571                 String a = p[0];
572                 String st = p.length == 2 ? p[1] : null;
573                 try {
574                     out.write("<pre>");
575                     AssetTypeDef def = mgr.findByName(a, st);
576                     if (def != null) {
577                         generator.printAssetTypeDef(def);
578                     }
579                 } catch (Exception e) {
580                     out.println(e.getMessage() + " on " + a + "/" + st);
581                     e.printStackTrace(new PrintWriter(out));
582                 }
583                 out.write("</pre>");
584 
585             }
586 
587         }
588 
589         for (String assetType : mgr.getAssetTypes()) {
590             for (String subtype : mgr.getSubTypes(assetType)) {
591                 // %><input type="checkbox" name="assettype"
592                 // value="<%=assetType%>:<%=subtype%>" /><%=assetType%> /
593                 // <%=subtype%><br/><%
594             }
595             try {
596                 AssetTypeDef def = mgr.findByName(assetType, null);
597                 if (!def.getProperties().getIsFlexAsset()) {
598                     // %><input type="checkbox" name="assettype"
599                     // value="<%=assetType%>" /><%=assetType%><br/><%
600                 }
601             } catch (Exception e) {
602                 out.println(e.getMessage());
603                 DebugHelper.printStackTrace(out, e);
604             }
605 
606         }
607     }
