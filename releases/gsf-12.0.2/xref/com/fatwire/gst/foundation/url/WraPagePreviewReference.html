43  public class WraPagePreviewReference extends PageRef {
44  
45      public static final String GST_DISPATCHER = "GST/Dispatcher";
46      
47  
48      @SuppressWarnings({ "rawtypes", "unchecked" })
49      @Override
50      public void setParameters(Map args, ICS ics) throws ReferenceException {
51  
52          // no processing to do if not serving a page for SS
53          if (shouldModify(args, ics)) {
54              AssetId id = new AssetIdImpl((String) args.get("c"), Long.parseLong((String) args.get("cid")));
55              AssetApiVirtualWebrootDao vwDao = new AssetApiVirtualWebrootDao(ics);
56              WraCoreFieldDao wraDao = new AssetApiWraCoreFieldDao(ics);
57              String currentEnvironment = vwDao.getVirtualWebrootEnvironment();
58              // only look up webroots for WRAs when the environment is configured
59              if (currentEnvironment != null && wraDao.isVanityAsset(id)) {
60                   VanityAsset wra = wraDao.getVanityWra(id);
61                  // get the webroot
62                  VirtualWebroot vw = vwDao.lookupVirtualWebrootForAsset(wra);
63                  if (vw != null) {
64                      // set the special fields
65                      args.put("virtual-webroot", vw.getEnvironmentVirtualWebroot());
66                      args.put("url-path", wra.getPath().substring(vw.getMasterVirtualWebroot().length()));
67                      // has pagename been set? if not, use default.
68                      String pagename = ics.GetProperty(WraPathAssembler.DISPATCHER_PROPNAME,
69                              "ServletRequest.properties", true);
70                      if (!goodString(pagename)) {
71                          pagename = GST_DISPATCHER;
72                      }
73                      // pagename or wrapperpage depending on whether or not we're
74                      // going to use a wrapper.
75                      if (args.get(PubConstants.WRAPPERPAGE) != null)
76                          args.put(PubConstants.WRAPPERPAGE, pagename);
77                      else
78                          args.put("pagename", pagename);
79                  } else {
80                      if (log.isDebugEnabled()) {
81                          log.debug("Not adding WRAPathAssembler args because no matching virtual webroot found for path "
82                                  + wra.getPath() + " and environemnt " + currentEnvironment);
83                      }
84                  }
85              } else {
86                  if (log.isDebugEnabled()) {
87                      if (currentEnvironment == null) {
88                          log.debug("Not adding WraPathAssembler args because virtual webroot environment is not configured");
89                      } else {
90                          log.debug("Not adding WraPathAssembler args because asset " + id + " is not web referenceable.");
91                      }
92                  }
93              }
94          } else {
95              if (log.isDebugEnabled()) {
96                  log.debug("Not adding WRAPathAssembler args because context is not satellite server (it is "
97                          + getSatelliteContext() + ").  Args: " + args);
98              }
99          }
100         super.setParameters(args, ics);
101     }
102 
103     protected boolean shouldModify(Map<String, String> args, ICS ics) {
104         return isRendermodeLive(ics) && getSatelliteContext() == SatelliteContext.SATELLITE_SERVER
105                 && isGetTemplateUrl(args);
106     }
107 
108     protected boolean isRendermodeLive(ICS ics) {
109         String rendermode = ics.GetVar(PubConstants.RENDERMODE);
110         return StringUtils.isBlank(rendermode) || "live".equals(rendermode);
111     }
112 
113     /**
114      * Check to see if the tag being called is a getTemplateUrl tag. If it is
115      * not, we should not be processing this for special links. Note it's not
116      * that easy to figure this out, and there could be missing pieces here.
117      * 
118      * @param args tag args
119      * @return true if it's a gettemplateurl tag, false otherwise.
120      */
121     private boolean isGetTemplateUrl(Map<String, String> args) {
122         if (args.get("c") == null)
123             return false;
124         if (args.get("cid") == null)
125             return false;
126         String pagename = args.get(ftMessage.PageName);
127         if (pagename == null)
128             return false;
129         if (pagename.split("/").length < 2)
130             return false; // need site/type/tname or site/tname at least for a
131                           // valid URL
132         if (args.get(PubConstants.WRAPPERPAGE) != null)
133             return true; // wrapper is only supported for GTU calls
134         else {
135             // possible further checks here just in case
136         }
137         return true;
138     }
