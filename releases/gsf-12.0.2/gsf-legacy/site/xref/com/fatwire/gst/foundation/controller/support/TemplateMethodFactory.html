16  package com.fatwire.gst.foundation.controller.support;
17  
18  import java.lang.reflect.Constructor;
19  import java.lang.reflect.InvocationTargetException;
20  import java.lang.reflect.Method;
21  
22  /**
23   * @author Dolf Dijkstra
24   * @deprecated helper class; no replacement
25   */
26  public class TemplateMethodFactory {
27      private static final Class<?>[] NO_PARAMS = new Class[0];
28      private static final Object[] NO_ARGS = new Object[0];
29  
30      @SuppressWarnings("unchecked")
31      public static <T> T createByMethod(Object template, Class<T> c) throws SecurityException, NoSuchMethodException,
32              IllegalArgumentException, IllegalAccessException, InvocationTargetException {
33          Method m;
34          m = template.getClass().getMethod("create" + c.getSimpleName(), NO_PARAMS);
35          if (m != null) {
36              Object o = m.invoke(template, NO_ARGS);
37              if (o != null && c.isAssignableFrom(o.getClass())) {
38                  return (T) o;
39  
40              }
41          }
42          return null;
43      }
44  
45      static <T> T createByConstructor(Class<T> c) throws SecurityException, NoSuchMethodException, IllegalArgumentException, InstantiationException, IllegalAccessException, InvocationTargetException   {
46          if(c.isInterface()) return null; //TODO medium test for abstract class
47          final Constructor<T> constr = c.getConstructor(NO_PARAMS);
48          return constr.newInstance();
49      }
50  
51  }
