20  package tools.gsf.facade.runtag.asset;
21  
22  import COM.FutureTense.Interfaces.ICS;
23  import tools.gsf.facade.runtag.TagRunner;
24  
25  import java.util.ArrayList;
26  import java.util.Enumeration;
27  import java.util.List;
28  
29  /**
30   * @author Dolf Dijkstra
31   */
32  
33  public class LoadAndExport implements TagRunner {
34  
35      private final String assetType;
36  
37      private final long assetId;
38  
39      public LoadAndExport(final String assetType, final long assetId) {
40          super();
41          this.assetType = assetType;
42          this.assetId = assetId;
43      }
44  
45      public String execute(final ICS ics) {
46          final String name = "asset" + ics.genID(true);
47          try {
48              final AssetLoadById al = new AssetLoadById();
49              al.setName(name);
50              al.setAssetType(assetType);
51              al.setAssetId(assetId);
52              // al.setEditable(true);
53              al.setOption(AssetLoadById.OPTION_READ_ONLY_COMPLETE);
54  
55              al.execute(ics);
56  
57              new AssetScatter(name, "as", "PubList").execute(ics);
58              new AssetScatter(name, "as", true).execute(ics);
59              new AssetExport(name, "as", "xml").execute(ics);
60  
61              final String xml = ics.GetVar("xml");
62              ics.RemoveVar("xml");
63              return xml;
64          } finally {
65              // cleaning up
66              ics.SetObj(name, null);// clear obj from ics
67              final List<String> toClean = new ArrayList<String>();
68              for (final Enumeration<?> e = ics.GetVars(); e.hasMoreElements(); ) {
69                  final String k = (String) e.nextElement();
70                  if (k.startsWith("as:")) {
71                      toClean.add(k);
72                  }
73              }
74              // preventing java.util.ConcurrentModificationException
75              for (String n : toClean) {
76                  ics.RemoveVar(n);
77              }
78  
79          }
80      }
81  
82      public long getAssetId() {
83          return assetId;
84      }
85  
86      public String getAssetType() {
87          return assetType;
88      }
89  
90  }
