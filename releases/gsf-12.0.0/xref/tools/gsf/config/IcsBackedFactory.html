43  /**
44   * @author Tony Field
45   * @since 2016-08-05
46   */
47  public class IcsBackedFactory extends AbstractDelegatingFactory<ICS> {
48  
49      private final ICS ics;
50      
51      protected ICS getICS() {
52      	return this.ics;
53      }
54  
55      public IcsBackedFactory(ICS ics, Factory delegate) {
56          super(ics, delegate);
57          this.ics = ics;
58      }
59  
60      @ServiceProducer(cache = true, name="bindInjector")
61      public Injector createBindInjector() {
62          return new BindInjector(ics);
63      }
64  
65      @ServiceProducer(cache = true, name="injectForRequestInjector")
66      public Injector createInjectForRequestInjector() {
67          Factory factory = FactoryLocator.locateFactory(ics);
68          return new InjectForRequestInjector(factory);
69      }
70  
71      @ServiceProducer(cache = true)
72      public MappingService createMappingService() {
73          AssetAccessTemplate aat = getObject("assetAccessTemplate", AssetAccessTemplate.class);
74          return new IcsMappingService(ics, aat);
75      }
76  
77      @ServiceProducer(cache = true, name="mappingInjector")
78      public Injector createMappingInjector(final ICS ics) {
79          MappingService mappingService = getObject("mappingService", MappingService.class);
80          return new MappingInjector(ics, mappingService);
81      }
82  
83      @ServiceProducer(cache = true, name="currentAssetInjector")
84      public Injector createCurrentAssetInjector(final ICS ics) {
85          AssetAccessTemplate aat = getObject("assetAccessTemplate", AssetAccessTemplate.class);
86          TemplateAssetAccess taa = getObject("templateAssetAccess", TemplateAssetAccess.class);
87          ScatteredAssetAccessTemplate saa = getObject("scatteredAssetAccessTemplate", ScatteredAssetAccessTemplate.class);
88          return new CurrentAssetInjector(ics, taa, saa, aat);
89      }
90  
91      @ServiceProducer(cache = true, name="compositeInjector")
92      public Injector createCompositeInjector() {
93          Injector bind = getObject("bindInjector", Injector.class);
94          Injector map = getObject("mappingInjector", Injector.class);
95          Injector ifr = getObject("injectForRequestInjector", Injector.class);
96          Injector currentAsset = getObject("currentAssetInjector", Injector.class);
97          Stopwatch stopwatch = getObject("stopwatch", Stopwatch.class);
98          return new AnnotationInjector(stopwatch, bind, map, ifr, currentAsset);
99      }
100 
101     @ServiceProducer(cache = true)
102     public PropertyDao createPropertyDao(final ICS ics) {
103     	Session session = SessionFactory.getSession(ics);
104     	AssetDataManager adm = (AssetDataManager) session.getManager(AssetDataManager.class.getName());
105     	SiteManager sm = (SiteManager) session.getManager(SiteManager.class.getName());
106     	String type = "GSTProperty";
107     	String flexDefName = "GSTProperty";
108     	String propNameAttr = "name";
109     	String propDescAttr = "description";
110     	String propValueAttr = "value";
111     	return new AssetApiPropertyDao(adm, sm, type, flexDefName, propNameAttr, propDescAttr, propValueAttr, ics);
112     }
113 
114     @ServiceProducer(cache = true)
115     public AssetAccessTemplate createAssetAccessTemplate() {
116         return new AssetAccessTemplate(this.ics);
117     }
118     
119     @ServiceProducer(cache = true)
120     public ScatteredAssetAccessTemplate createScatteredAssetAccessTemplate() {
121         return new ScatteredAssetAccessTemplate(this.ics);
122     }
123 
124     @ServiceProducer(cache = true)
125     public TemplateAssetAccess createTemplateAssetAccess() {
126         return new TemplateAssetAccess(this.ics);
127     }
128     
129     @ServiceProducer(cache = true)
130     public LocaleService createLocaleService(final ICS ics) {
131         return new DefaultLocaleService(ics);
132     }
133     
134 }
