16  package com.fatwire.gst.foundation.controller.action;
17  
18  import javax.servlet.ServletContext;
19  import javax.servlet.http.HttpServletResponse;
20  
21  import COM.FutureTense.Interfaces.ICS;
22  import COM.FutureTense.Util.ftErrors;
23  
24  import com.fatwire.gst.foundation.CSRuntimeException;
25  import com.fatwire.gst.foundation.controller.AbstractController;
26  import com.fatwire.gst.foundation.facade.RenderUtils;
27  
28  /**
29   * Dispatching Controller. Relies on ActionLocator to dispatch control to Action
30   * classes. An ActionNameResolver is used to resolve the action name on the
31   * current request.
32   * 
33   * @author Dolf Dijkstra
34   * @since May 26, 2011
35   * 
36   * @deprecated as of release 12.x, replace GSF Actions with WCS 12c's native Controllers and/or wrappers
37   * 
38   */
39  public abstract class AbstractActionController extends AbstractController {
40  
41      public AbstractActionController(ICS ics) {
42          super(ics);
43      }
44  
45      @Override
46      public final void doExecute() {
47  
48          // record seid and eid
49          RenderUtils.recordBaseCompositionalDependencies(ics);
50  
51          // find the action locator
52          LOG.trace("Dispatcher looking for action locator");
53          final ActionLocator locator = getActionLocator();
54          if (locator == null)
55              throw new IllegalStateException("No ActionLocator returned by class " + getClass().getName());
56          if (LOG.isTraceEnabled()) {
57              LOG.trace("Using action locator: " + locator.getClass().getName());
58          }
59          ActionNameResolver resolver = getActionNameResolver();
60          if (resolver == null)
61              throw new IllegalStateException("No ActionNameResolver returned by class " + getClass().getName());
62          String actionName = resolver.resolveActionName(ics);
63          // get the action
64          final Action action = locator.getAction(ics, actionName);
65          if (LOG.isTraceEnabled()) {
66              LOG.trace("Using action: " + action.getClass().getName());
67          }
68          if (action == null)
69              throw new IllegalStateException(
70                      "No Action found. An ActionLocator should always return a Action. ActionLocation "
71                              + locator.getClass().getName() + " did not return an action for '" + actionName + "'.");
72  
73          // execute the command
74          action.handleRequest(ics);
75          LOG.trace("Request handling complete");
76      }
77  
78      /**
79       * @return the ActionNameResolver
80       */
81      protected abstract ActionNameResolver getActionNameResolver();
82  
83      /**
84       * @return the ActionLocator.
85       */
86      protected abstract ActionLocator getActionLocator();
87  
88      @SuppressWarnings("deprecation")
89      protected ServletContext getServletContext() {
90          return ics.getIServlet().getServlet().getServletContext();
91      }
92  
93      @Override
94      public final void handleException(final Exception e) {
95          if (LOG.isTraceEnabled()) {
96              // Give developer a clue in case error pages aren't configured
97              // properly.
98              LOG.trace("Action threw an exception and an error code will be returned.  Exception: " + e, e);
99          }
100         if (e instanceof CSRuntimeException) {
101             handleCSRuntimeException((CSRuntimeException) e);
102         } else {
103             sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e);
104         }
105     }
106 
107     /**
108      * This method transforms errno values into http status codes and sets them
109      * using the X-Fatwire-Status header.
110      * <p>
111      * Only some errnos are handled by this base class.
112      * <p>
113      * More info coming soon
114      * 
115      * @param e exception
116      */
117     protected final void handleCSRuntimeException(final CSRuntimeException e) {
118         switch (e.getErrno()) {
119             case HttpServletResponse.SC_BAD_REQUEST:
120             case ftErrors.badparams:
121                 sendError(HttpServletResponse.SC_BAD_REQUEST, e);
122                 break;
123             case HttpServletResponse.SC_NOT_FOUND:
124             case ftErrors.pagenotfound:
125                 sendError(HttpServletResponse.SC_NOT_FOUND, e);
126                 break;
127             case HttpServletResponse.SC_FORBIDDEN:
128             case ftErrors.noprivs:
129                 sendError(HttpServletResponse.SC_FORBIDDEN, e);
130                 break;
131             default:
132                 sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e);
133                 break;
134         }
135     }
