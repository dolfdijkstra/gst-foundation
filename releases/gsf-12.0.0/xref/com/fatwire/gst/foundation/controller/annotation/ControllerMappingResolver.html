17  package com.fatwire.gst.foundation.controller.annotation;
18  
19  import java.lang.reflect.Method;
20  
21  import COM.FutureTense.Interfaces.ICS;
22  
23  /**
24   * This class resolves a specific method that handles a specific request as a
25   * controller. The method needs to be annotated with the IcsVariable annotation
26   * and need to accept one argument of type ICS.
27   * <p>
28   * The name of the method can be freely chosen. The method needs to have public
29   * visibility.
30   * 
31   * @author Dolf Dijkstra
32   * @since Mar 21, 2011
33   * @deprecated - WCS 12c supports its own native controller infrastructure
34   */
35  public class ControllerMappingResolver {
36  
37      /**
38       * @param ics Content Server context object
39       * @param o object with method annotations of type IcsVariable
40       * @return the method with the IcsVariable annotation that has a match for
41       *         the name value pair as set in the annotation.
42       */
43      public Method findControllerMethod(final ICS ics, final Object o) {
44          for (final Method m : o.getClass().getMethods()) {
45              final IcsVariable p = m.getAnnotation(IcsVariable.class);
46              if (p != null) {
47                  for (final String param : p.var()) {
48                      final String[] split = param.split("=");
49                      if (split[1].equals(ics.GetVar(split[0]))) {
50  
51                          if (m.getParameterTypes().length == 1 && m.getParameterTypes()[0].equals(ICS.class)) {
52                              return m;
53                          }
54                          throw new UnsupportedOperationException("Method " + m.getName()
55                                  + " does not have a single argument of type ICS though the method is annotated "
56                                  + "with a IcsVariable annation.");
57  
58                      }
59                  }
60              }
61  
62          }
63          return null;
64      }
65  }
