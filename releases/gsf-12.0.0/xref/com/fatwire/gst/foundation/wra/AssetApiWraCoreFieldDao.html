94      /**
95       * Method to test whether or not an asset is web-referenceable.
96       * 
97       * @param id asset ID to check
98       * @return true if the asset is a valid web-referenceable asset, false if it
99       *         is not
100      */
101     public boolean isWebReferenceable(AssetId id) {
102         try {
103             WebReferenceableAsset wra = getWra(id);
104             return isWebReferenceable(wra);
105         } catch (RuntimeException e) {
106             return false;
107         }
108     }

114     /**
115      * Method to test whether or not an asset is web-referenceable. 
116      *
117      * @param candidate asset to test
118      * @return true if the asset is a valid web-referenceable asset, false if it
119      *         is not
120      */
121     public boolean isWebReferenceable(WebReferenceableAsset candidate) {
122         return StringUtils.isNotBlank(candidate.getPath()) && StringUtils.isNotBlank(candidate.getTemplate());

138     private AssetMapper<WebReferenceableAsset> mapper = new AssetMapper<WebReferenceableAsset>() {
139 
140         public WebReferenceableAsset map(AssetData data) {
141             WraBeanImpl wra = new WraBeanImpl();
142             wra.setId(data.getAssetId());
143             wra.setName(AttributeDataUtils.getWithFallback(data, "name"));
144             wra.setDescription(AttributeDataUtils.asString(data.getAttributeData("description")));
145             wra.setSubtype(AttributeDataUtils.asString(data.getAttributeData("subtype")));
146             wra.setStatus(AttributeDataUtils.asString(data.getAttributeData("status")));
147             wra.setStartDate(AttributeDataUtils.asDate(data.getAttributeData("startdate")));
148             wra.setEndDate(AttributeDataUtils.asDate(data.getAttributeData("enddate")));
149             wra.setMetaTitle(AttributeDataUtils.getWithFallback(data, "metatitle"));
150             wra.setMetaDescription(AttributeDataUtils.getWithFallback(data, "metadescription"));
151             wra.setMetaKeyword(AttributeDataUtils.asString(data.getAttributeData("metakeyword")));
152             wra.setH1Title(AttributeDataUtils.getWithFallback(data, "h1title"));
153             wra.setLinkText(AttributeDataUtils.getWithFallback(data, "linktext", "h1title"));
154             wra.setPath(AttributeDataUtils.asString(data.getAttributeData("path")));
155             wra.setTemplate(AttributeDataUtils.asString(data.getAttributeData("template")));
156             return wra;
157         }
158

178     private static final String ASSETPUBLICATION_QRY = "SELECT p.name from Publication p, AssetPublication ap "
179             + "WHERE ap.assettype = ? " + "AND ap.assetid = ? " + "AND ap.pubid=p.id";
180     static final PreparedStmt AP_STMT = new PreparedStmt(ASSETPUBLICATION_QRY,
181             Collections.singletonList("AssetPublication")); // todo: low
182                                                             // priority:
183     // determine
184     // why publication
185     // cannot fit there.
186 
187     static {
188         AP_STMT.setElement(0, "AssetPublication", "assettype");
189         AP_STMT.setElement(1, "AssetPublication", "assetid");
190     }
191 
192     public String resolveSite(String c, String cid) {
193         final StatementParam param = AP_STMT.newParam();
194         param.setString(0, c);
195         param.setLong(1, Long.parseLong(cid));
196         String result = null;
197         for (Row pubid : SqlHelper.select(ics, AP_STMT, param)) {
198             if (result != null) {
199                 LOG.warn("Found asset "
200                         + c
201                         + ":"
202                         + cid
203                         + " in more than one publication. It should not be shared; aliases are to be used for cross-site sharing.  Controller will use first site found: "
204                         + result);
205             } else {
206                 result = pubid.getString("name");
207             }
208         }
209         return result;
210     }

212     static final PreparedStmt FIND_P = new PreparedStmt(
213             "SELECT art.oid FROM AssetRelationTree art, AssetPublication ap, Publication p WHERE ap.assetid = art.oid "
214                     + "AND ap.assettype = 'Page' AND ap.pubid = p.id AND p.name = ? "
215                     + "AND art.otype = ap.assettype AND art.nid in "
216                     + "(SELECT nparentid FROM AssetRelationtree WHERE otype=? AND oid=? AND ncode='-') ORDER BY ap.id",
217             Arrays.asList("AssetRelationTree", "AssetPublication", "Publication"));
218 
219     static {
220         FIND_P.setElement(0, "Publication", "name");
221         FIND_P.setElement(1, "AssetRelationTree", "otype");
222         FIND_P.setElement(2, "AssetRelationTree", "oid");
223     }

225     /**
226      * Locate the page that contains the specified Web-Referenceable Asset.
227      * <p>
228      * A WRA is supposed to just be placed on one page (in the unnamed
229      * association block), and this method locates it. If it is not found, 0L is
230      * returned.
231      * <p>
232      * If multiple matches are found, a warning is logged and the first one is
233      * returned.
234      * 
235      * @param wraAssetIdWithSite id of the web-referenceable asset. Site is
236      *            included
237      * @return page asset ID or 0L.
238      */
239     public long findP(AssetIdWithSite wraAssetIdWithSite) {
240         ics.ClearErrno();
241         final StatementParam param = FIND_P.newParam();
242         param.setString(0, wraAssetIdWithSite.getSite());
243         param.setString(1, wraAssetIdWithSite.getType());
244         param.setLong(2, wraAssetIdWithSite.getId());
245         long result = 0L;
246         for (final Row r : SqlHelper.select(ics, FIND_P, param)) {
247             if (result != 0L)
248                 LOG.warn("Asset "
249                         + wraAssetIdWithSite
250                         + " was found as the primary content on multiple pages in the site.  Web-referenceable assets should only be the primary content on one Page to comply with SEO rules.  Automatically selecting the oldest page");
251             else
252                 result = r.getLong("oid");
253         }
254         if (result == 0L) {
255             // could not locate
256         }
257         return result;
258 
259     }

261     @Override
262     public boolean isVanityAsset(AssetId id) {
263         try {
264             AssetData data = AssetDataUtils.getAssetData(ics, id, "path", "template");
265             String path = AttributeDataUtils.asString(data.getAttributeData("path"));
266             String template = AttributeDataUtils.asString(data.getAttributeData("template"));
267             return StringUtils.isNotBlank(path) && StringUtils.isNotBlank(template);
268         } catch (RuntimeException e) {
269             return false;
270         }
271     }

273     @Override
274     public VanityAsset getVanityWra(AssetId id) {
275         AssetData data = AssetDataUtils.getAssetData(ics, id, VANITY_ATTRIBUTE_NAMES);
276         return vmapper.map(data);
277     }
