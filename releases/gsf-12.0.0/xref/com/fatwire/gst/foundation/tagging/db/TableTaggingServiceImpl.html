84      public void install() {
85          TableDef def = new TableDef(TAGREGISTRY_TABLE, TABLE_ACL_LIST, "obj");
86          // todo: low priority: define the PK properly (this will work for now)
87  
88          def.addColumn("tag", TableColumn.Type.ccvarchar).setLength(255).setNullable(false);
89          def.addColumn("assettype", TableColumn.Type.ccvarchar).setLength(255).setNullable(false);
90          def.addColumn("assetid", TableColumn.Type.ccbigint).setNullable(false);
91          def.addColumn("startdate", TableColumn.Type.ccdatetime).setNullable(true);
92          def.addColumn("enddate", TableColumn.Type.ccdatetime).setNullable(true);
93  
94          new TableCreator(ics).createTable(def);
95      }

98          return SqlHelper.tableExists(ics, TAGREGISTRY_TABLE);

101     public void recordCacheDependency(Tag tag) {
102         CacheManager.RecordItem(ics, convertTagToCacheDepString(tag));
103     }

105     public void clearCacheForTag(Collection<Tag> tags) {
106         List<String> ids = new ArrayList<String>();
107         for (Tag tag : tags) {
108             ids.add(convertTagToCacheDepString(tag));
109         }
110         CacheManager cm = new CacheManager(ics);
111         cm.setPagesByID(ics, ids.toArray(new String[ids.size()]));
112         cm.flushCSEngine(ics, CacheHelper._both);
113         cm.flushSSEngines(ics);
114     }

116     public void addAsset(AssetId id) {
117         // Only add assets that are tagged.
118 
119         TaggedAsset asset = loadTaggedAsset(id);
120         if (isTagged(asset)) {
121             if (LOG.isTraceEnabled()) {
122                 LOG.trace("Adding tagged asset to tag registry: " + asset);
123             }
124             for (Tag tag : asset.getTags()) {
125 
126                 AddRow r = new AddRow(TAGREGISTRY_TABLE);
127 
128                 r.set("id", ics.genID(true));
129                 r.set("tag", tag.getTag());
130                 r.set("assettype", id.getType());
131                 r.set("assetid", id.getId());
132                 r.set("startdate", asset.getStartDate());
133                 r.set("enddate", asset.getEndDate());
134                 r.execute(ics);
135             }
136         }
137     }

145     public void deleteAsset(AssetId id) {
146         if (LOG.isTraceEnabled()) {
147             LOG.trace("Attempting to remove asset from tag registry:" + id);
148         }
149         SqlHelper.execute(
150                 ics,
151                 TAGREGISTRY_TABLE,
152                 "DELETE FROM " + TAGREGISTRY_TABLE + " WHERE assettype = '" + id.getType() + "' and assetid = "
153                         + id.getId());
154         if (LOG.isDebugEnabled()) {
155             LOG.debug("Deleted tagged asset " + id + " from tag registry (or asset was never there in the first place)");
156         }
157     }

159     public Collection<Tag> getTags(AssetId id) {
160         // this method records all the compositional dependencies that we need
161         // for this method (this is critical)
162         return loadTaggedAsset(id).getTags();
163     }

165     public Collection<Tag> getTags(Collection<AssetId> ids) {
166         HashSet<Tag> tags = new HashSet<Tag>();
167         // todo: medium priority: IMPORTANT: This should be optimized so that we
168         // don't kill the db
169         if (ids.size() > 5)
170             LOG.warn("Fetching tags serially for " + ids.size()
171                     + " assets.  TableTaggingServiceImpl isn't yet optimized to handle this very nicely");
172         for (AssetId id : ids) {
173             tags.addAll(getTags(id));
174         }
175         return tags;
176     }

178     /**
179      * Retrieve the tags for the tagged asset. This method records a
180      * compositional dependency on both the input asset AND the tags themselves.
181      * 
182      * @param id asset id
183      * @return tagged asset
184      */
185     private TaggedAsset loadTaggedAsset(AssetId id) {
186         LogDep.logDep(ics, id);
187 
188         // Temporarily disable usage of asset APIs in this use case due to a bug
189         // in which asset listeners
190         // cause a deadlock when the asset API is used.
191 
192         final TaggedAsset ret;
193         final String gsttagAttrVal;
194         if (directSqlAccessTools.isFlex(id)) {
195             PreparedStmt basicFields = new PreparedStmt("SELECT id,startdate,enddate" + " FROM " + id.getType()
196                     + " WHERE id = ?", Collections.singletonList(id.getType()));
197             basicFields.setElement(0, id.getType(), "id");
198 
199             StatementParam param = basicFields.newParam();
200             param.setLong(0, id.getId());
201             Row row = SqlHelper.selectSingle(ics, basicFields, param);
202 
203             Date start = StringUtils.isBlank(row.getString("startdate")) ? null : row.getDate("startdate");
204             Date end = StringUtils.isBlank(row.getString("enddate")) ? null : row.getDate("enddate");
205             ret = new TaggedAsset(id, start, end);
206             // todo: medium: optimize as this is very inefficient for flex
207             // assets
208             gsttagAttrVal = directSqlAccessTools.getFlexAttributeValue(id, "gsttag");
209 
210         } else {
211 
212             PreparedStmt basicFields = new PreparedStmt("SELECT * FROM " + id.getType() + " WHERE ID = ?",
213                     Collections.singletonList(id.getType()));
214             basicFields.setElement(0, id.getType(), "id");
215 
216             StatementParam param = basicFields.newParam();
217             param.setLong(0, id.getId());
218             Row row = SqlHelper.selectSingle(ics, basicFields, param);
219 
220             Date start = StringUtils.isBlank(row.getString("startdate")) ? null : row.getDate("startdate");
221             Date end = StringUtils.isBlank(row.getString("enddate")) ? null : row.getDate("enddate");
222             ret = new TaggedAsset(id, start, end);
223             String s = "";
224             try {
225                 if (row.isField("gsttag")) {
226                     s = row.getString("gsttag");
227                 }
228             } catch (Exception e) {
229                 LOG.trace("Could not get gsttag data from basic asset.  Maybe this is just because "
230                         + "there is no gsttag column - which is just fine.", e);
231             }
232             gsttagAttrVal = s;
233         }
234 
235         if (StringUtils.isNotBlank(gsttagAttrVal)) {
236             for (String tag : gsttagAttrVal.split(",")) {
237                 Tag oTag = asTag(tag);
238                 recordCacheDependency(oTag);
239                 ret.addTag(oTag);
240             }
241         }
242 
243         // End temporary deadlock workaround
244 
245         if (LOG.isTraceEnabled())
246             LOG.trace("Loaded tagged asset " + ret);
247         return ret;
248     }

250     private AssetMapper<TaggedAsset> mapper = new AssetMapper<TaggedAsset>() {
251 
252         public TaggedAsset map(AssetData data) {
253             Date startDate = AttributeDataUtils.asDate(data.getAttributeData("startdate"));
254             Date endDate = AttributeDataUtils.asDate(data.getAttributeData("enddate"));
255             TaggedAsset ret = new TaggedAsset(data.getAssetId(), startDate, endDate);
256             for (String tag : AttributeDataUtils.getAndSplitString(data.getAttributeData("gsttag"), ",")) {
257                 Tag oTag = asTag(tag);
258 
259                 ret.addTag(oTag);
260             }
261 
262             return ret;
263         }
264 
265     };

268      * Retrieve the tags for the tagged asset. This method records a
269      * compositional dependency on both the input asset AND the tags themselves.

271      * @param id asset id
272      * @return tagged asset
273      */

275     @SuppressWarnings("unused")
276     private TaggedAsset loadTaggedAssetNew(AssetId id) {
277         LogDep.logDep(ics, id);
278         AssetAccessTemplate aat = new AssetAccessTemplate(ics);
279         TaggedAsset ret = aat.readAsset(id, mapper, "startdate", "enddate", "gsttag");
280 
281         for (Tag tag : ret.getTags()) {
282             recordCacheDependency(tag);
283         }
284         if (LOG.isTraceEnabled())
285             LOG.trace("Loaded tagged asset " + ret);
286         return ret;
287     }

289     public Collection<AssetId> lookupTaggedAssets(Tag tag) {
290         recordCacheDependency(tag);
291         final StatementParam param = REGISTRY_SELECT.newParam();
292         param.setString(0, tag.getTag());
293         List<AssetId> ids = new ArrayList<AssetId>();
294         for (final Row asset : SqlHelper.select(ics, REGISTRY_SELECT, param)) {
295             AssetId id = new AssetIdImpl(asset.getString("assettype"), asset.getLong("assetid"));
296             LogDep.logDep(ics, id);
297             if (FilterAssetsByDate.isValidOnDate(ics, id, null)) {
298                 ids.add(id);
299             } else {
300                 if (LOG.isDebugEnabled())
301                     LOG.debug("Asset " + id + " tagged with " + tag + " is not active based on startdate/enddate");
302             }
303         }
304         return ids;
305     }

307     public boolean isTagged(AssetId id) {
308         TaggedAsset ta = loadTaggedAsset(id);
309         return isTagged(ta);
310     }

312     public boolean isTagged(TaggedAsset ta) {
313         if (ta == null)
314             return false;
315         try {
316             if (ta.getTags().size() > 0) {
317                 if (LOG.isTraceEnabled()) {
318                     LOG.trace("isTagged loaded the asset and found that " + ta.getId() + " is a tagged asset.");
319                 }
320                 return true;
321             } else {
322                 if (LOG.isTraceEnabled()) {
323                     LOG.trace("isTagged loaded the asset and found that " + ta.getId() + " is not a tagged asset.");
324                 }
325                 return false;
326             }
327         } catch (RuntimeException e) {
328             if (LOG.isTraceEnabled()) {
329                 LOG.trace(
330                         "isTagged found that " + ta.getId() + " is not a tagged asset.  We found an exception: " + e.toString(),
331                         e);
332             }
333             return false;
334         }
335     }

337     private static final PreparedStmt REGISTRY_SELECT = new PreparedStmt(
338             "SELECT tag, assettype, assetid, startdate, enddate " + "FROM " + TAGREGISTRY_TABLE
339                     + " WHERE tag=? ORDER BY startdate,enddate", Collections.singletonList(TAGREGISTRY_TABLE));

341     static {
342         REGISTRY_SELECT.setElement(0, TAGREGISTRY_TABLE, "tag");
343     }

345     private static class TaggedAsset {
346         final AssetId id;
347         final Date startDate;
348         final Date endDate;
349         final Collection<Tag> tags;
350 
351         TaggedAsset(AssetId id, Date startDate, Date endDate) {
352             this.id = id;
353             this.startDate = startDate;
354             this.endDate = endDate;
355             tags = new ArrayList<Tag>();
356         }
357 
358         public AssetId getId() {
359             return id;
360         }
361 
362         public Date getStartDate() {
363             return startDate;
364         }
365 
366         public Date getEndDate() {
367             return endDate;
368         }
369 
370         public Collection<Tag> getTags() {
371             return tags;
372         }
373 
374         private void addTag(Tag tag) {
375             tags.add(tag);
376         }
377 
378         public String toString() {
379             return id.toString() + "|startdate:" + startDate + "|enddate:" + endDate + "|tags:" + tags;
380         }
381     }
