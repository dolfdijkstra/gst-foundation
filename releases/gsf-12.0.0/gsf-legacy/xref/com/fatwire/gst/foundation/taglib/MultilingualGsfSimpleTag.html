39  /**
40   * simple tag for translating an asset
41   *
42   * @author Tony Field
43   * @since 2011-11-28
44   * 
45   * 
46   * @deprecated as of release 12.x
47   * 
48   */
49  public abstract class MultilingualGsfSimpleTag extends GsfSimpleTag {
50      protected static final Logger LOG = LoggerFactory.getLogger("tools.gsf.legacy.taglib.MultilingualGsfSimpleTag");
51  
52      private String dimensionSetName = null;
53      private long dimensionSetId = -1L;
54      private String localeName = null;
55      private long localeId = -1L;
56  
57      public final void setDimset(String s) {
58          try {
59              dimensionSetId = Long.parseLong(s);
60          } catch (NumberFormatException e) {
61              dimensionSetName = s;
62              dimensionSetId = -1L;
63          }
64      }
65  
66      public final void setLocale(String s) {
67          try {
68              localeId = Long.parseLong(s);
69          } catch (NumberFormatException e) {
70              localeName = s;
71              localeId = -1L;
72          }
73      }
74  
75      /**
76       * Return a dimension filter instance corresponding to the dimension set specified by the user (or discovered by
77       * the tag).  The dimension filter is configured with the preferred dimensions of the user (also onfigured).
78       *
79       * The preferred locales are identified by checking the following locations, in the order specified:
80       * 1) set by the locale attribute by id of locale
81       * 2) set by locale attribute by name of locale
82       * 3) detected by finding the locale dimension id in the ics variable "locale"
83       * 4) detected by finding the locale name in the ics variable "locale"
84       * 5) detected by finding the locale dimension id in the ics session variable "locale"
85       * 6) detected by finding the locale name in the ics session variable "locale"
86       * 7) detected by reading the Accept-Language header
87       *
88       * The dimension set is identified by checking in the following places, in order:
89       * 1) set by the dimset attribute by name of dimension set
90       * 2) set by dimset attribute by the id of the dimension set
91       * 3) looked up by finding the site name in the ics variable "site" and loading the single dimension set associated with that site
92       *
93       * @return a dimension filter, configured with the set preferred locales, or null, if either the dimenion set or the
94       * preferred dimensions could not be found (with extensive errors)
95       */
96      protected final DimensionFilterInstance getDimensionFilter() {
97          ICS ics = getICS();
98          DimensionFilterInstance filter;
99          try {
100             Collection<AssetId> preferredLocales = getPreferredLocales();
101             DimensionSetInstance dimSet = getDimensionSet();
102             filter = DimensionUtils.getDimensionFilter(DimensionUtils.getDM(ics), preferredLocales, dimSet);
103             if (LOG.isDebugEnabled()) {
104                 LOG.debug("Multilingual-enabled tag located dimension filter: " + filter + " in dimensionSet " + dimSet + " with preferred locales: " + preferredLocales + " ");
105             }
106         } catch (DimensionException e) {
107             LOG.error("Multilingual-enabled tag could not locate dimension filter", e);
108             filter = null;
109         } catch (RuntimeException e) {
110             LOG.error("Multilingual-enabled tag could not locate dimension filter", e);
111             filter = null;
112         }
113         return filter;
114     }
115 
116     /**
117      * Get the locale that the user explicitly specified.  If not set, null is returned.
118      * @return the id of the locale that the user explicitly set.  Handles setting by name or assetid.
119      */
120     protected final AssetId getExplicitlySpecifiedLocale() {
121         ICS ics = getICS();
122         // first, check for explicitly specified by ID
123         if (localeId != -1L) {
124             DimensionManager dm = DimensionUtils.getDM(ics);
125             Dimension d = dm.loadDimension(localeId);
126             if (d != null) {
127                 LOG.trace("Preferred locale explicitly set to " + localeId);
128                 return d.getId();
129             }
130         }
131 
132         // next, check for explicitly specified by name
133         if (localeName != null) {
134             Dimension d = DimensionUtils.getDimensionForName(ics, localeName);
135             if (d != null) {
136                 LOG.trace("Preferred locale explicitly set to " + localeName);
137                 return d.getId();
138             }
139         }
140         return null;
141     }
142     /**
143      * Get the ordered list of preferred locales that the user wants.  Multiple attempts are made to figure out the right locale.
144      * @return collection of asset identifiers of the preferred locales 
145      */
146     protected final Collection<AssetId> getPreferredLocales() {
147         AssetId result = getExplicitlySpecifiedLocale();
148         if (result != null) return Collections.singleton(result);
149 
150         ICS ics = getICS();
151 
152         // next, check for implicitly specified by ID using locale variable
153         String localeVar = getICS().GetVar("locale");
154         try {
155             long localeIdFromVar = Long.parseLong(localeVar);
156             DimensionManager dm = DimensionUtils.getDM(ics);
157             Dimension d = dm.loadDimension(localeIdFromVar);
158             if (d != null) {
159                 LOG.trace("Preferred locale detected in ICS context using 'locale' variable: " + localeIdFromVar);
160                 return Collections.singletonList(d.getId());
161             }
162         } catch (NumberFormatException e) {
163             // maybe it's a locale name...
164             try {
165                 Dimension d = DimensionUtils.getDimensionForName(ics, localeVar);
166                 if (d != null) {
167                     LOG.trace("Preferred locale detected in ICS context using 'locale' variable: " + localeVar);
168                     return Collections.singletonList(d.getId());
169                 }
170             } catch (Exception ex) {
171                 // nope... don't worry, we'll find it....
172             }
173         }
174 
175         // next, check for implicitly specified by ID using locale session variable
176         String localeSSVar = getICS().GetSSVar("locale");
177         try {
178             long localeIdFromSSVar = Long.parseLong(localeSSVar);
179             DimensionManager dm = DimensionUtils.getDM(ics);
180             Dimension d = dm.loadDimension(localeIdFromSSVar);
181             if (d != null) {
182                 LOG.trace("Preferred locale detected in ICS context using 'locale' session variable: " + localeIdFromSSVar);
183                 return Collections.singletonList(d.getId());
184             }
185         } catch (NumberFormatException e) {
186             // maybe it's a locale name...
187             try {
188                 Dimension d = DimensionUtils.getDimensionForName(ics, localeSSVar);
189                 if (d != null) {
190                     LOG.trace("Preferred locale detected in ICS context using 'locale' session variable: " + localeSSVar);
191                     return Collections.singletonList(d.getId());
192                 }
193             } catch (Exception ex) {
194                 // nope... don't worry, we'll find it....
195             }
196         }
197 
198         // finally, get the locale from the servlet request's Accept-Language header..
199         List<AssetId> preferredLocales = new ArrayList<AssetId>();
200         @SuppressWarnings({ "rawtypes", "deprecation" })
201         Enumeration locales = ics.getIServlet().getServletRequest().getLocales();
202         while (locales.hasMoreElements()) {
203             Locale locale = (Locale) locales.nextElement();
204             if (locale != null) {
205                 String localeName = locale.toString();
206                 if (localeName != null && localeName.length() > 0) {
207                     try {
208                         Dimension dimension = DimensionUtils.getDimensionForName(ics, localeName);
209                         preferredLocales.add(dimension.getId());
210                         LOG.trace("Found registered locale in user's Accept-Language header (or default): " + localeName);
211                     } catch (RuntimeException e) {
212                         // don't care if the dimension is not in the system - they probably won't all be there
213                         // and we're guessing anyway, so it's okay.
214                         LOG.trace("Found a locale in the user's Accept-Language header, but it was not registered as a dimension: " + localeName + " (this is not usually an error)", e);
215                     }
216                 }
217             }
218         }
219         return preferredLocales;
220     }
221 
222     private final DimensionSetInstance getDimensionSet() {
223         if (dimensionSetName != null) {
224             return LocaleUtils.getDimensionSet(getICS(), dimensionSetName);
225         }
226         if (dimensionSetId != -1L) {
227             return LocaleUtils.getDimensionSet(getICS(), dimensionSetId);
228         }
229         try {
230             ICS ics = getICS();
231             String site = ics.GetVar("site");
232             if (site != null && site.length() > 0) {
233                 long discoveredId = LocaleUtils.locateDimensionSetForSite(ics, site);
234                 LOG.trace("Auto-discovered dimension set because there is only one in site " + site + ": DimensionSet:" + discoveredId);
235                 return LocaleUtils.getDimensionSet(ics, discoveredId);
236             }
237         } catch (RuntimeException e) {
238             LOG.trace("Could not auto-discovered dimensionset: " + e);
239         }
240         throw new IllegalArgumentException("DimensionSet not found");
241     }
