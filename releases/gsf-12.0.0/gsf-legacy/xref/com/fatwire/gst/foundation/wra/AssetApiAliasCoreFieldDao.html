38  /**
39   * Dao for dealing with core fields in an alias. Aliases may override fields in
40   * their target WRA if it points to another asset. Aliases may also refer to
41   * external URLs.
42   * 
43   * @author Tony Field
44   * @since Jul 21, 2010
45   * 
46   * @deprecated as of release 12.x, replace with WCS 12c's native vanity URLs support.
47   * 
48   */
49  public class AssetApiAliasCoreFieldDao implements AliasCoreFieldDao {
50  
51      public static final String TARGET_ASSOCIATION_NAME = "target";
52      private final ICS ics;
53      private final WraCoreFieldDao wraCoreFieldDao;
54  
55      public AssetApiAliasCoreFieldDao(ICS ics, WraCoreFieldDao wraCoreFieldDao) {
56          this.ics = ics;
57          this.wraCoreFieldDao = wraCoreFieldDao;
58      }
59  
60      private static final Logger LOG = LoggerFactory.getLogger("tools.gsf.legacy.wra.AssetApiAliasCoreFieldDao");
61  
62      /**
63       * Return an AssetData object containing the core fields found in an alias
64       * asset.
65       * <p>
66       * Also includes selected metadata fields:
67       * <ul>
68       * <li>id</li>
69       * <li>name</li>
70       * <li>subtype</li>
71       * <li>startdate</li>
72       * <li>enddate</li>
73       * <li>status</li>
74       * </ul>
75       * 
76       * @param id id of alias asset
77       * @return AssetData containing core fields for Alias asset
78       */
79      public AssetData getAsAssetData(AssetId id) {
80          return AssetDataUtils.getAssetData(ics, id); // load all instead of
81                                                       // specific
82          // fields, because we do not
83          // know which fields are set.
84      }
85  
86      /**
87       * Method to test whether or not an asset is an Alias. todo: low priority:
88       * optimize as this will be called at runtime
89       * 
90       * @param id asset ID to check
91       * @return true if the asset is a valid Alias asset, false if it is not
92       */
93      public boolean isAlias(AssetId id) {
94          if (Alias.ALIAS_ASSET_TYPE_NAME.equals(id.getType()) == false)
95              return false;
96          try {
97              getAlias(id);
98              return true;
99          } catch (RuntimeException e) {
100             return false;
101         }
102     }
103 
104     /**
105      * Return an alias asset bean given an input id. Required fields must be set
106      * or an exception is thrown.
107      * 
108      * @param id asset id
109      * @return Alias
110      * @see #isAlias
111      */
112     public Alias getAlias(AssetId id) {
113         TemplateAsset alias = new TemplateAsset(getAsAssetData(id));
114         AssetId target = Children.getOptionalSingleAssociation(ics, id.getType(), Long.toString(id.getId()),
115                 TARGET_ASSOCIATION_NAME);
116         if (target == null) {
117             String targetUrl = alias.asString("target_url");
118             if (targetUrl != null && targetUrl.length() == 0)
119                 throw new IllegalStateException(
120                         "Asset is not an alias because it has neither a target_url attribute nor a target named association: "
121                                 + id);
122             LOG.trace("Alias " + id + " refers to an external URL");
123 
124             AliasBeanImpl o = new AliasBeanImpl();
125             // Wra fields
126             o.setId(id);
127             o.setName(alias.asString("name"));
128             o.setDescription(alias.asString("description"));
129             o.setSubtype("GSTAlias");
130             o.setStatus(alias.asString("status"));
131             o.setStartDate(alias.asDate("startdate"));
132             o.setEndDate(alias.asDate("enddate"));
133             String linktext = alias.asString("linktext");
134             o.setLinkText(goodString(linktext) ? linktext : o.getH1Title());
135 
136             // these fields really don't make much sense in an external link...
137             if (alias.isAttribute("h1title"))
138                 o.setH1Title(alias.asString("h1title"));
139             if (alias.isAttribute("metatitle"))
140                 o.setMetaTitle(alias.asString("metatitle"));
141             if (alias.isAttribute("metadescription"))
142                 o.setMetaDescription(alias.asString("metadescription"));
143             if (alias.isAttribute("metakeyword"))
144                 o.setMetaKeyword(alias.asString("metakeyword"));
145             o.setPath(alias.asString("path"));
146             o.setTemplate(alias.asString("template"));
147 
148             // Alias fields
149             o.setTargetUrl(targetUrl);
150             if (alias.isAttribute("popup"))
151                 o.setPopup(alias.asString("popup"));
152             if (alias.isAttribute("linkimage"))
153                 o.setLinkImage(alias.asAssetId("linkimage"));
154             return o;
155 
156         } else {
157             if (!wraCoreFieldDao.isWebReferenceable(target)) {
158                 throw new IllegalStateException(
159                         "Asset is not a valid alias because it refers to a target asset that is not web-referenceable. Alias:"
160                                 + id + ", target:" + target);
161             }
162             WebReferenceableAsset wra = wraCoreFieldDao.getWra(target);
163             LOG.trace("Alias " + id + " refers to another wra asset: " + target);
164 
165             AliasBeanImpl o = new AliasBeanImpl();
166             // Wra fields
167             o.setId(id);
168             if (LOG.isTraceEnabled())
169                 LOG.trace("alias name: " + alias.asString("name") + ", wra name:" + wra.getName());
170             o.setName(alias.asString("name"));
171             if (LOG.isTraceEnabled())
172                 LOG.trace("alias description: " + alias.asString("description") + ", wra description:"
173                         + wra.getDescription());
174             o.setDescription(alias.asString("description"));
175             o.setSubtype("GSTAlias");
176             if (LOG.isTraceEnabled())
177                 LOG.trace("alias status: " + alias.asString("status") + ", wra status:" + wra.getStatus());
178             o.setStatus(alias.asString("status"));
179 
180             Date d = alias.asDate("startdate");
181             o.setStartDate(d == null ? wra.getStartDate() : d);
182             d = alias.asDate("enddate");
183             o.setEndDate(d == null ? wra.getEndDate() : d);
184 
185             if (alias.isAttribute("h1title")) {
186                 String s = alias.asString("h1title");
187                 o.setH1Title(s == null ? wra.getH1Title() : s);
188             }
189 
190             if (alias.isAttribute("linktext")) {
191                 String s = alias.asString("linktext");
192                 o.setLinkText(s == null ? wra.getLinkText() : s);
193             }
194 
195             if (alias.isAttribute("metatitle")) {
196                 String s = alias.asString("metatitle");
197                 if (!goodString(s))
198                     s = wra.getMetaTitle();
199                 o.setMetaTitle(s);
200             }
201 
202             if (alias.isAttribute("metadescription")) {
203                 String s = alias.asString("metadescription");
204                 if (!goodString(s))
205                     s = wra.getMetaDescription();
206                 o.setMetaDescription(s);
207             }
208 
209             if (alias.isAttribute("metakeyword")) {
210                 String s = alias.asString("metakeyword");
211                 if (!goodString(s))
212                     s = wra.getMetaKeyword();
213                 o.setMetaKeyword(s);
214             }
215 
216             String s = alias.asString("path");
217             if (LOG.isTraceEnabled())
218                 LOG.trace("alias path: " + s + ", wra path:" + wra.getPath());
219             if (!goodString(s))
220                 s = wra.getPath();
221             o.setPath(s);
222 
223             s = alias.asString("template");
224             if (LOG.isTraceEnabled())
225                 LOG.trace("alias template: " + s + ", wra template:" + wra.getTemplate());
226             if (!goodString(s))
227                 s = wra.getTemplate();
228             o.setTemplate(s);
229 
230             // Alias fields
231             o.setTarget(target);
232             if (alias.isAttribute("popup"))
233                 o.setPopup(alias.asString("popup"));
234             if (alias.isAttribute("linkimage"))
235                 o.setLinkImage(alias.asAssetId("linkimage"));
236 
237             return o;
238         }
239 
240     }

242     private static final String ASSETPUBLICATION_QRY = "SELECT p.name from Publication p, AssetPublication ap "
243             + "WHERE ap.assettype = ? " + "AND ap.assetid = ? " + "AND ap.pubid=p.id";
244     static final PreparedStmt AP_STMT = new PreparedStmt(ASSETPUBLICATION_QRY,
245             Collections.singletonList("AssetPublication")); // todo: low
246                                                             // priority:
247                                                             // determine
248     // why publication
249     // cannot fit there.
250 
251     static {
252         AP_STMT.setElement(0, "AssetPublication", "assettype");
253         AP_STMT.setElement(1, "AssetPublication", "assetid");
254     }
255 
256     public String resolveSite(String c, String cid) {
257         final StatementParam param = AP_STMT.newParam();
258         param.setString(0, c);
259         param.setLong(1, Long.parseLong(cid));
260         String result = null;
261         for (Row pubid : SqlHelper.select(ics, AP_STMT, param)) {
262             if (result != null) {
263                 LOG.warn("Found asset "
264                         + c
265                         + ":"
266                         + cid
267                         + " in more than one publication. It should not be shared; aliases are to be used for cross-site sharing.  Controller will use first site found: "
268                         + result);
269             } else {
270                 result = pubid.getString("name");
271             }
272         }
273         return result;
274     }
