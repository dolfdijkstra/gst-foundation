57      public VirtualWebroot getVirtualWebroot(long cid) {
58          String sCid = Long.toString(cid);
59          if (LOG.isTraceEnabled())
60              LOG.trace("Loading virtual webroot data for for GSTVirtualWebroot:" + sCid);
61          AssetData ad = AssetDataUtils.getAssetData(ics, AssetIdUtils.createAssetId("GSTVirtualWebroot", sCid), "master_vwebroot", "env_vwebroot",
62                  "env_name");
63          return new VWebrootBeanImpl(cid, AttributeDataUtils.getWithFallback(ad, "master_vwebroot"),
64                  AttributeDataUtils.getWithFallback(ad, "env_vwebroot"), AttributeDataUtils.getWithFallback(ad,
65                          "env_name"));
66  
67      }

69      /**
70       * Get all of the virtual webroots, sorted by URL length.
71       * 
72       * @return list of virtual webroots
73       */
74      public SortedSet<VirtualWebroot> getAllVirtualWebroots() {
75          AssetList al = new AssetList();
76          al.setExcludeVoided(true);
77          al.setList("pr-out");
78          al.setType("GSTVirtualWebroot");
79          ics.RegisterList("pr-out", null);
80          al.execute(ics);
81          IList ilist = ics.GetList("pr-out");
82          ics.RegisterList("pr-out", null);
83          if (ilist == null)
84              throw new IllegalStateException("No GSTVirtualWebroots are registered");
85  
86          SortedSet<VirtualWebroot> result = new TreeSet<VirtualWebroot>(new UrlInfoComparator());
87          for (Row r : new IListIterable(ilist)) {
88              result.add(getVirtualWebroot(r.getLong("id")));
89          }
90          return result;
91      }

93      /**
94       * Get the current virtual webroot environment as defined by the
95       * configuration properties. Null indicates that none is configured.
96       * 
97       * @return virtual webroot environment or null if not set.
98       */
99      public String getVirtualWebrootEnvironment() {
100         String environmentName = System.getProperty("com.fatwire.gst.foundation.env-name", null);
101 
102         // avoid configuration problem trickery
103         if (environmentName != null) {
104             environmentName = environmentName.trim();
105             if (environmentName.length() == 0)
106                 environmentName = null;
107         }
108 
109         if (environmentName == null) {
110             // allow user to have accidentally mis-configured things
111             environmentName = ics.GetProperty("com.fatwire.gst.foundation.env-name");
112             // avoid configuration problem trickery
113             if (environmentName != null) {
114                 environmentName = environmentName.trim();
115                 if (environmentName.length() == 0)
116                     environmentName = null;
117             }
118         }
119         if (environmentName == null)
120             LOG.debug("Virtual webroot environment is not configured.");
121         return environmentName;
122     }

124     /**
125      * Look up and return the VirtualWebroot corresponding to the specified
126      * WebReferenceableAsset, for the current environment. If the current
127      * environment is not configured, no match can be found.
128      * 
129      * @param wra web-referenceable asset
130      * @return matching VirtualWebroot or null if no match is found.
131      */
132     public VirtualWebroot lookupVirtualWebrootForAsset(VanityAsset wra) {
133         if (LOG.isDebugEnabled())
134             LOG.debug("Looking up virtual webroot for WRA " + wra.getId());
135         String wraPath = wra.getPath();
136         return lookupVirtualWebrootForUri(wraPath);
137     }
138 
139     public VirtualWebroot lookupVirtualWebrootForUri(String wraPath) {
140         if (wraPath == null) {
141             LOG.trace("WRA does not have a path set - cannot locate virtual webroot");
142             return null;
143         }
144         String env = getVirtualWebrootEnvironment();
145         if (env == null)
146             return null;
147         for (VirtualWebroot vw : getAllVirtualWebroots()) {
148             // find longest first one that is found in the prefix of path. that
149             // is virtual-webroot
150             // the path in the asset must start with the MASTER virtual webroot
151             // for this to work. This could
152             // be loosened up but there is no real reason to right now.
153             if (env.equals(vw.getEnvironmentName()) && wraPath.startsWith(vw.getMasterVirtualWebroot())) {
154                 return vw;
155             }
156         }
157         return null; // no match
158     }

160     /**
161      * Comparator that compares virtual webroots by webroot. Uses
162      * reverse-natural ordering to ensure that overlapping virtual webroots
163      * resolve properly.
164      */
165     public static class UrlInfoComparator implements Comparator<VirtualWebroot> {
166 
167         public int compare(VirtualWebroot o1, VirtualWebroot o2) {
168             int i = -o1.getMasterVirtualWebroot().compareTo(o2.getMasterVirtualWebroot());
169             if (i == 0) {
170                 int j = -o1.getEnvironmentName().compareTo(o2.getEnvironmentName());
171                 if (j == 0) {
172                     int k = -o1.getEnvironmentVirtualWebroot().compareTo(o2.getEnvironmentVirtualWebroot());
173                     if (k == 0) {
174                         return (int) (o1.getId().getId() - o2.getId().getId());
175                     }
176                     return k;
177                 }
178                 return j;
179             }
180             return i;
181         }
182     }
