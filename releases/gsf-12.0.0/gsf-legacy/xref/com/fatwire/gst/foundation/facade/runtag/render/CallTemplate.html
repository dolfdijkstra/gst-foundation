39  /**
40   * CallTemplate tag with many improvements around context and style.
41   * <p>
42   * <code>
43   * &lt;RENDER.CALLTEMPLATE SITE="site name"
44   * SLOTNAME="name of slot"
45   * TID="caller Template or CSElement id" [TTYPE="caller Template or CSElement"]
46   * [C="asset type"] [CID="asset id"] [TNAME="target Template or CSElement name"]
47   * [CONTEXT="context override"] [STYLE="pagelet or element"]
48   * [VARIANT="template variant name"] [PACKEDARGS="packed arguments"]&gt;
49   * 
50   * [&lt;RENDER.ARGUMENT NAME="variable1" VALUE="value1"/&gt;]
51   * 
52   * &lt;/RENDER.CALLTEMPLATE&gt;
53   * </code>
54   * 
55   * <p>
56   * <b>
57   * IMPORTANT: this LEGACY render:calltemplate facade defaults to style="element", amongst other scenarios: 1) If you are rendering
58   *            an "uncached" template from another "uncached" pagelet and you have not explicitly set style to something other
59   *            than "element" (i.e. "pagelet" or "embedded") or 2) If the cid for the called Template is the same as that of the
60   *            current (callee) Template.
61   *            
62   *            In WCS 12c, calling an uncached Template with render.calltemplate using style="element" implies any Controller
63   *            attached to the called Template WILL NOT GET INVOKED. That, in turn, may result in unexpected behaviour or, most
64   *            likely, broken code.
65   * </b> 
66   * </p>
67   * 
68   * @author Tony Field
69   * @author Dolf Dijkstra
70   * @since Jun 10, 2010
71   * @deprecated - com.fatwire.gst.foundation.facade and all subpackages have moved to the tools.gsf.facade package
72   */

123     @Override
124     protected void preExecute(final ICS ics) {
125         // consider to set site to ics.GetVar("site"); as helper for reasonable
126 
127         // default value
128         readConfig(ics);
129 
130         if (defaultStyle != null) {
131             setStyle(defaultStyle);
132         } else if (override || style == null) {
133             final Style newStyle = proposeStyle(ics);
134             setStyle(newStyle);
135         }
136         ics.ClearErrno();
137         super.preExecute(ics);
138     }

205     void readConfig(final ICS ics) {
206         if (configLoaded) {
207             return;
208         }
209         final String tmp = getProperty(ics, "style");
210         if (tmp != null) {
211             CallTemplate.defaultStyle = Style.valueOf(tmp);
212         }
213         CallTemplate.override = "true".equals(getProperty(ics, "override"));
214         CallTemplate.config_FixPageCriteria = "true".equals(getProperty(ics, "config_FixPageCriteria"));
215         CallTemplate.configLoaded = true;
216 
217     }

219     private String getProperty(final ICS ics, final String name) {
220         String val = System.getProperty(CallTemplate.class.getName() + "." + name);
221         if (!Utilities.goodString(val)) {
222             val = ics.GetProperty(CallTemplate.class.getName() + "." + name, "futuretense_xcel.ini", true);
223         }
224         LOG.trace(CallTemplate.class.getName() + "." + name + "=" + val);
225         return val;
226     }

228     public Style proposeStyle(final ICS ics) {
229 
230         /**
231          * Considerations 1) Check target for parameter callstyle and use that
232          * 
233          */
234         String pname = getTargetPagename();
235 
236         // String targetStyle =(String)
237         // ics.getPageData(pname).getDefaultArguments().get("callstyle");
238         final boolean targetCached = RenderUtils.isCacheable(ics, pname);
239         final boolean currentCached = RenderUtils.isCacheable(ics, ics.GetVar(ftMessage.PageName));
240 
241         final Style proposal = calculateStyle(ics, pname, currentCached, targetCached);
242         if (LOG.isDebugEnabled()) {
243             LOG.debug("Setting style to '" + proposal + (style != null ? "' (user did set '" + style + "')" : "'")
244                     + " for calltemplate to '" + pname + "' with " + type + "," + cid + "," + getList()
245                     + " in element: '" + ics.ResolveVariables("CS.elementname") + "', caching: '" + currentCached + "/"
246                     + targetCached + "', page: " + ics.pageURL());
247         }
248         return proposal;
249 
250     }

252     private String getTargetPagename() {
253         String pname;
254 
255         if (tname.startsWith("/")) // typeless
256         {
257             pname = site + tname;
258         } else {
259             pname = site + "/" + type + "/" + tname;
260         }
261         return pname;
262     }

264     private Style calculateStyle(final ICS ics, final String pname, final boolean currentCache,
265             final boolean targetCache) {
266         if (currentCache == false) // we are not caching for the current pagelet
267         {
268             if (targetCache == false) {
269                 return Style.element; // call as element is target is also not (WARNING: in WCS 12c, this means Controller does not get invoked)
270                 // cacheable
271             } else {
272                 checkPageCriteria(ics, pname);
273                 return Style.pagelet; // otherwise call as pagelet
274             }
275 
276         } else { // currently we are caching
277 
278             if (targetCache == false) {
279                 checkPageCriteria(ics, pname);
280                 return Style.pagelet;
281             } else {
282                 // LOG.debug("getvar.cid=" + ics.GetVar("cid") + " at " +
283                 // ics.pageURL());
284 
285                 final FTValList m = COM.FutureTense.Interfaces.Utilities.getParams(ics.pageURL());
286                 final String pageCid = m.getValString("cid");
287                 if (pageCid != null && !pageCid.equals(ics.GetVar("cid"))) {
288                     LOG.warn(ics.GetVar("cid") + " does not match cid (" + pageCid + ") in " + ics.pageURL());
289                 }
290                 // should we check if cid is current page criteria, we are a
291                 // Template??
292                 if (cid != null && cid.equals(pageCid)) {
293                     // if c/cid does not change than we call this as an element,
294                     // as reuse is unlikely
295                     if (LOG.isTraceEnabled()) {
296                         LOG.trace("Calling " + pname + " as an element from " + ics.ResolveVariables("CS.elementname")
297                                 + " because cid is same as on current pagelet.");
298                     }
299                     return Style.element; // (WARNING: in WCS 12c, this means Controller does not get invoked)
300                 } else {
301                     checkPageCriteria(ics, pname);
302                     return Style.embedded; // this is calltemplate, assuming
303                     // that
304                     // headers/footers/leftnavs etc will be
305                     // via CSElements/SiteEntry
306                 }
307             }
308 
309         }

312     protected static final List<String> CALLTEMPLATE_EXCLUDE_VARS = Collections.unmodifiableList(Arrays.asList("TNAME",
313             "C", "CID", "EID", "SEID", "PACKEDARGS", "VARIANT", "CONTEXT", "SITE", "TID", "rendermode", "ft_ss",
314             "SystemAssetsRoot", "cshttp", "errno", "tablename", "empty", "errdetail", "null"));
315 
316     @SuppressWarnings("unchecked")
317     private void checkPageCriteria(final ICS ics, final String target) {
318         final FTValList o = getList();
319 
320         if (o != null) {
321             String[] pc = ics.pageCriteriaKeys(target);
322             if (pc == null) {
323                 pc = new String[0];
324             }
325             final Map<String, ?> m = o;
326             for (final Iterator<?> i = m.entrySet().iterator(); i.hasNext();) {
327                 final Entry<String, ?> e = (Entry<String, ?>) i.next();
328                 final String key = e.getKey();
329                 // only inspect arguments that start with ARGS_
330                 if (key.startsWith(ARGS)) {
331 
332                     String shortKey = key.substring(ARGS.length());
333                     boolean found = CALLTEMPLATE_EXCLUDE_VARS.contains(shortKey);
334                     if (!found) {
335                         for (final String c : pc) {
336                             if (c.equalsIgnoreCase(shortKey)) {
337                                 found = true;
338                                 break;
339                             }
340                         }
341                     }
342                     if (!found) {
343                         LOG.error("Argument '" + key + "' not found as PageCriterium on " + target
344                                 + ". Calling element is " + ics.ResolveVariables("CS.elementname")
345                                 + ". Arguments are: " + m.keySet().toString() + ". PageCriteria: " + Arrays.asList(pc),
346                                 new Exception());
347                         // we could correct this by calling as an element
348                         // or by removing the argument
349                         if (isFixPageCriteria() || config_FixPageCriteria) {
350                             i.remove();
351                             LOG.warn("Argument '" + key + "' is removed from the call to '" + target
352                                     + "' as it is not a PageCriterium.");
353                         }
354 
355                     }
356                 }
357             }
358 
359         }

363     protected void handleError(ICS ics) {
364 
365         int errno = ics.GetErrno();
366         if (errno != -10004) // error checking was too aggressive, any error set
367                              // by an element would leak into this handler
368             return;
369         FTValList arguments = list;
370         ftErrors complexError = ics.getComplexError();
371         String pagename = ics.GetVar("pagename");
372         String elementname = ics.ResolveVariables("CS.elementname");
373         String msg = "ics.runTag(RENDER.CALLTEMPLATE) failed for tname: " + list.getValString("TNAME") + " for asset: "
374                 + list.getValString("C") + ":" + list.getValString("CID") + " within pagename:" + pagename;
375         if (elementname != null)
376             msg += " and element:" + elementname;
377         msg += ".";
378         ics.ClearErrno();
379         throw new TagRunnerRuntimeException(msg, errno, arguments, complexError, pagename, elementname);
380     }
