54      /**
55       * Get all of the parent assets in the AssetRelationTree for the specified
56       * asset. Association name is required, but expectedParentType is an
57       * optional filter argument.
58       * <p>
59       * Does not record any asset dependencies
60       *
61       * @param ics                ICS context
62       * @param log                logger. May be null.
63       * @param child              child asset id
64       * @param expectedParentType asset type of the parent to be returned. If
65       *                           null, type of parent is irrelevant.
66       * @param associationName    name of association to use while looking for
67       *                           parents. may not be null
68       * @return list of parents, never null.
69       * @see #getParents(ICS, AssetId, String[])
70       */
71      public static List<AssetId> getAssetRelationTreeParents(ICS ics, Logger log, AssetId child, String expectedParentType, String associationName) {
72          FTValList vl = new FTValList();
73          vl.setValString("ftcmd", "findnode");
74          vl.setValString("treename", "AssetRelationTree");
75          vl.setValString("where", "oid");
76          vl.setValString("oid", Long.toString(child.getId()));
77          if (ics.TreeManager(vl)) {
78              int errno = ics.GetErrno();
79              if (errno < 0) {
80                  switch (errno) {
81                      case -111:
82                          if (log != null && log.isTraceEnabled()) {
83                              log.trace("Node not found in AssetRelationTree for asset " + child);
84                          }
85                          return Collections.emptyList();
86                      default: {
87                          throw new CSRuntimeException("Failed to look up asset " + child + " in AssetRelationTree.", errno);
88                      }
89                  }
90              }
91  
92              // found node. get its parent
93              IList art = ics.GetList("AssetRelationTree");
94              ics.RegisterList("AssetRelationTree", null);
95  
96              List<AssetId> parents = new ArrayList<AssetId>();
97  
98              if (art == null || !art.hasData() || art.numRows() == 0) {
99                  if (log != null && log.isTraceEnabled()) {
100                     log.trace("Failed to locate " + child + " in AssetRelationTree.");
101                 }
102             } else {
103 
104                 List<String> childNodeIds = new ArrayList<String>();
105                 for (IList row : new IterableIListWrapper(art)) {
106                     if (child.getType().equals(IListUtils.getStringValue(row, "otype"))) {
107                         String nid = IListUtils.getStringValue(row, "nid");
108                         String ncode = IListUtils.getStringValue(row, "ncode");
109                         if (log != null && log.isTraceEnabled()) {
110                             log.trace("Found " + child + " in AssetRelationTree.  Node ID: " + nid + ", ncode: " + ncode + ", expecting ncode: " + associationName);
111                         }
112                         if (associationName.equals(ncode)) {
113                             childNodeIds.add(IListUtils.getStringValue(row, "nid"));
114                         }
115                     }
116                 }
117 
118                 for (String nid : childNodeIds) {
119                     vl.clear();
120                     vl.setValString("ftcmd", "getparent");
121                     vl.setValString("treename", "AssetRelationTree");
122                     vl.setValString("node", nid);
123                     if (ics.TreeManager(vl) && ics.GetErrno() >= 0) {
124                         art = ics.GetList("AssetRelationTree");
125                         ics.RegisterList("AssetRelationTree", null);
126                         AssetId parent = new AssetIdImpl(IListUtils.getStringValue(art, "otype"), Long.valueOf(IListUtils.getStringValue(art, "oid")));
127                         if (log != null && log.isTraceEnabled()) {
128                             log.trace(child + " in AssetRelationTree has a parent " + parent);
129                         }
130                         if (expectedParentType == null) {
131                             parents.add(parent);
132                         } else {
133                             if (expectedParentType.equals(parent.getType())) {
134                                 parents.add(parent);
135                             } else {
136                                 if (log != null && log.isDebugEnabled()) {
137                                     log.debug("Parent " + parent + " is not of the expected type (" + expectedParentType + ") so it is being excluded from the return list for child: " + child);
138                                 }
139                             }
140                         }
141                     } else {
142                         throw new CSRuntimeException("Failed to look up parent of article " + child + " in AssetRelationTree.  TreeManager call failed unexpectedly", ics.GetErrno());
143                     }
144 
145                 }
146             }
147 
148             return parents;
149         } else {
150             throw new CSRuntimeException("Failed to look up article " + child + " in AssetRelationTree.  TreeManager call failed unexpectedly", ics.GetErrno());
151         }
152     }

154     /**
155      * Look up parents in Asset Relation Tree for the specified child.  Records asset dependencies as required (either a
156      * qualified or an unqualified unknowndeps).
157      *
158      * @param ics             context
159      * @param child           asset id that will have its parents retrieved
160      * @param associationName name of association to use for lookup.  May not ever be null.
161      * @return list of parents, never null.
162      */
163     public static Collection<AssetId> getParents(ICS ics, AssetId child, String... associationName) {
164 
165         // validate input
166         if (ics == null) {
167             throw new IllegalArgumentException("ICS cannot be null");
168         }
169         if (child == null) {
170             throw new IllegalArgumentException("Child asset id is required");
171         }
172         if (associationName == null) {
173             throw new IllegalArgumentException("Association name may not be null");
174         }
175 
176         List<String> assocNames = Arrays.asList(associationName); // so lame...
177 
178         Collection<AssetId> parents = new HashSet<AssetId>();
179 
180         for (Row childInfo : TreeHelper.findNode(ics, "AssetRelationTree", child)) {
181 
182             // right assoc name?
183             String ncode = childInfo.getString("ncode");
184             if (!assocNames.contains(ncode)) {
185                 if (LOG.isTraceEnabled()) {
186                     LOG.trace("Asset " + child + " with node " + childInfo.getString("nid") + " is not the child of any other asset using the association name " + assocNames + ". (This node is for the name " + ncode + ".)");
187                 }
188                 // nope...
189                 continue;
190             }
191 
192             // Yup. Find its parent.
193             for (Row parentInfo : TreeHelper.findParents(ics, "AssetRelationTree", childInfo.getString("nid"))) {
194                 AssetId parent = AssetIdUtils.createAssetId(parentInfo.getString("otype"), parentInfo.getString("oid"));
195                 if (LOG.isTraceEnabled()) {
196                     LOG.trace("Found parent " + parent + " of child " + child + " with association name " + ncode);
197                 }
198                 parents.add(parent);
199             }
200         }
201 
202         // log dep on child asset
203         LogDep.logDep(ics, child);
204 
205         // log dep on context
206         // todo: inspect the specified association definitions in "associationName" and only
207         // record asset-type specific deps instead of unknowndep if possible
208         Render.UnknownDeps(ics);
209 
210         if (LOG.isDebugEnabled()) {
211             LOG.debug("Looked up child asset " + child + " in AssetRelationTree for parents with the association names " + Arrays.asList(associationName) + " and found " + parents.size() + " results.  Details: " + (LOG.isTraceEnabled() ? parents : ""));
212         }
213 
214         return parents;
215     }
