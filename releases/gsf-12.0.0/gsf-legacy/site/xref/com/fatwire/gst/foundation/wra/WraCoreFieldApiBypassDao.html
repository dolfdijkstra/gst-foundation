81  	/**
82  	 * Method to test whether or not an asset is web-referenceable. todo: low
83  	 * priority: optimize as this will be called at runtime (assest api incache
84  	 * will mitigate the performance issue)
85  	 * 
86  	 * @param id asset id
87  	 *            asset ID to check
88  	 * @return true if the asset is a valid web-referenceable asset, false if it
89  	 *         is not
90  	 */
91  	@Override
92  	public boolean isWebReferenceable(final AssetId id) {
93  
94  		if (directSqlAccessTools.isFlex(id)) {
95  			if (!isWraEnabledFlexAssetType(id)) {
96  				return false;
97  			}
98  		} else {
99  			if (!isWraEnabledBasicAssetType(id)) {
100 				return false;
101 			}
102 
103 		}
104 		// type is wra, now lookup the asset data
105 		try {
106 			final WebReferenceableAsset wra = getWra(id);
107 			final boolean b = StringUtils.isNotBlank(wra.getPath());
108 			if (LOG.isTraceEnabled()) {
109 				LOG.trace("Asset "
110 						+ id
111 						+ (b ? " is " : " is not ")
112 						+ "web-referenceable, as determinted by the presence of a path attribute.");
113 			}
114 			return b;
115 		} catch (final RuntimeException e) {
116 			if (LOG.isTraceEnabled()) {
117 				LOG.trace("Asset " + id + " is not web-referenceable: " + e, e);
118 			}
119 			return false;
120 		}
121 	}

213 	/**
214 	 * Checks if the table definition for a basic asset has all the wra fields.
215 	 * 
216 	 * @param id asset id
217 	 */
218 	private boolean isWraEnabledBasicAssetType(final AssetId id) {
219 		boolean wraTable;
220 		final String listname = IListUtils.generateRandomListName();
221 		final IList list = ics.CatalogDef(id.getType(), listname,
222 				new StringBuffer());
223 		ics.RegisterList(listname, null);
224 		final List<String> attr = Arrays.asList(WRA_ATTRIBUTE_NAMES);
225 		int count = 0;
226 		for (final Row row : new IListIterable(list)) {
227 			/*
228 			 * "COLNAME" "COLTYPE" "COLSIZE" "KEY"
229 			 */
230 			if (attr.contains(row.getString("COLNAME").toLowerCase())) {
231 				count++;
232 			}
233 		}
234 		wraTable = count == attr.size();// all wra attributes are found in the
235 		// table def.
236 		if (LOG.isTraceEnabled()) {
237 			LOG.trace("Asset "
238 					+ id
239 					+ (wraTable ? " is " : " is not ")
240 					+ "web-referenceable, as determinted by the table definition.");
241 		}
242 
243 		return wraTable;
244 	}

246 	@Override
247 	public boolean isVanityAsset(AssetId id) {
248 		try {
249 			final PreparedStmt basicFields = new PreparedStmt(
250 					"SELECT path FROM " + id.getType() + " WHERE id = ?",
251 					Collections.singletonList(id.getType()));
252 			basicFields.setElement(0, id.getType(), "id");
253 
254 			final StatementParam param = basicFields.newParam();
255 			param.setLong(0, id.getId());
256 			final Row row = SqlHelper.selectSingle(ics, basicFields, param);
257 			if (row == null)
258 				return false;
259 			return StringUtils.isNotBlank(row.getString("path"));
260 		} catch (RuntimeException e) {
261 			return false;
262 		}
