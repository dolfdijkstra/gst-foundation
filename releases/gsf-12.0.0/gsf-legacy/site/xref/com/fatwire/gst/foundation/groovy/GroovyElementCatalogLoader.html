43  /**
44   * Loader for groovy script classes from the ElementCatalog
45   * 
46   * @author Dolf Dijkstra
47   * @since September 21,2012
48   * 
49   * @deprecated as of release 12.x, replace with WCS 12c's native Groovy support
50   * 
51   */
52  /*
53   * alternative method:
54   * http://groovy.codehaus.org/Alternate+Spring-Groovy-Integration
55   */
56  public class GroovyElementCatalogLoader extends DiskGroovyLoader {
57  	private PreparedStmt stmt;
58  	private Logger logger = LoggerFactory.getLogger("tools.gsf.legacy.groovy.GroovyElementCatalogLoader");
59  	private boolean isLoaded = false;
60  	private String path;
61  
62  	public GroovyElementCatalogLoader(ServletContext servletContext) {
63  		super(servletContext);
64  		/*
65  		 * ElementCatalog does not have Browser ACL, assumption here is that
66  		 * elements are updated as assets
67  		 */
68  		stmt = new PreparedStmt(
69  				"SELECT * FROM ElementCatalog WHERE elementname=?",
70  				Collections.singletonList("CSElement"));
71  		stmt.setElement(0, java.sql.Types.VARCHAR);
72  
73  	}
74  
75  	@Override
76  	public Object load(ICS ics, String resourceName) throws Exception {
77  		if (!isLoaded) {
78  			bootEngine(ics, path);
79  			isLoaded = true;
80  		}
81  
82  		if (logger.isDebugEnabled())
83  			logger.debug("Loading groovy script " + resourceName);
84  		if (ics.IsElement(resourceName)) {
85  
86  			final StatementParam param = stmt.newParam();
87  			param.setString(0, resourceName);
88  			Row row = SqlHelper.selectSingle(ics, stmt, param);
89  
90  			// ELEMENTNAME DESCRIPTION URL RESDETAILS1
91  			// RESDETAILS2
92  
93  			String url = row.getString("url");
94  			String res1 = row.getString("resdetails1");
95  			String res2 = row.getString("resdetails2");
96  			Map<String, String> m = new HashMap<String, String>();
97  			Utilities.getParams(res1, m, false);
98  			Utilities.getParams(res2, m, false);
99  			String tid = m.get("tid");
100 			String eid = m.get("eid");
101 			if (StringUtils.isNotBlank(tid)) {
102 				LogDep.logDep(ics, "Template", tid);
103 			}
104 			if (StringUtils.isNotBlank(eid)) {
105 				LogDep.logDep(ics, "CSElement", eid);
106 			}
107 			// prevent case where resourcename is same as a jsp element.
108 			if (url.endsWith(".groovy")) {
109 				// no loading based on fallback class name as the super method
110 				// has.
111 				if (logger.isDebugEnabled()) {
112 					logger.debug("Found element for " + resourceName + " => "
113 							+ url);
114 				}
115 				Class<?> x = getGroovyScriptEngine().loadScriptByName(url);
116 				return x.newInstance();
117 
118 			} else {
119 				return super.load(ics, resourceName);
120 			}
121 		} else {
122 			return super.load(ics, resourceName);
123 		}
124 
125 	}
126 
127 	@Override
128 	public void bootEngine(final String path) {
129 		this.path = path;
130 	}
131 
132 	public void bootEngine(ICS ics, final String path) {
133 		String[] root = new String[2];
134 
135 		String elementCatalogDefDir = ics
136 				.ResolveVariables("CS.CatalogDir.ElementCatalog");
137 		root[0] = elementCatalogDefDir;
138 		root[1] = path;
139 
140 		GroovyScriptEngine gse;
141 		try {
142 			gse = new GroovyScriptEngine(root, Thread.currentThread()
143 					.getContextClassLoader());
144 			gse.getConfig().setRecompileGroovySource(true);
145 			gse.getConfig().setMinimumRecompilationInterval(
146 					getMinimumRecompilationInterval());
147 			setGroovyScriptEngine(gse);
148 
149 		} catch (IOException e) {
150 			throw new RuntimeException(e);
151 		}
152 
153 	}
