99      public AssetId findTranslation(final AssetId id, final long preferredDimension, final long dimensionSetId) {
100         if (id == null) {
101             throw new IllegalArgumentException("Required Asset ID missing");
102         }
103 
104         if (_isInputAssetDimensionPreferred(id, preferredDimension)) {
105             _log.debug("Input dimension is already in the preferred dimension.  Not invoking dimension set filter.  Asset: "
106                     + id + ", dimension: " + preferredDimension);
107             return id;
108         } else {
109             _log.debug("About to look for translations.  Input asset id: " + id + ", dimension set: " + dimensionSetId
110                     + ", preferred dimension: " + preferredDimension);
111         }
112 
113         // *****************************************************************************
114         // The core business logic of this helper class is encapsulated in these
115         // 3 lines
116         final DimensionSetInstance dimset = _getDimensionSet(dimensionSetId);
117         return findTranslation(id, preferredDimension, dimset);
118 
119     }

129     public AssetId findTranslation(final AssetId id, final long preferredDimension, final String dimensionSetName) {
130         if (id == null) {
131             throw new IllegalArgumentException("Required Asset ID missing");
132         }
133         final Dimension locale = getLocaleForAsset(id);
134 
135         if (locale == null) {
136             _log.debug("Asset is not localized.  Not invoking dimension set filter.  Asset: " + id);
137             return id;
138         }
139         if (locale.getId().getId() == preferredDimension) {
140             _log.debug("Input dimension is already in the preferred dimension.  Not invoking dimension set filter.  Asset: "
141                     + id + ", dimension: " + preferredDimension);
142             return id;
143         } else {
144             _log.debug("About to look for translations.  Input asset id: " + id + ", dimension set: "
145                     + dimensionSetName + ", preferred dimension: " + preferredDimension);
146         }
147 
148         // *****************************************************************************
149         // The core business logic of this helper class is encapsulated in these
150         // 3 lines
151         final DimensionSetInstance dimset = getDimensionSet(dimensionSetName);
152 
153         return findTranslation(id, preferredDimension, dimset);
154     }

165     public AssetId findTranslation(final AssetId id, final long preferredDimension, final DimensionSetInstance dimset) {
166 
167         final DimensionFilterInstance filter = _getPopulatedDimensionFilter(dimset, preferredDimension);
168         // Get the relatives using the appropriate filter
169         final Collection<AssetId> relatives = getDAM().getRelatives(id, filter, "Locale");
170         // *****************************************************************************
171 
172         // make the result pretty
173         if (relatives == null) {
174             _log.warn("No translation found for asset " + id + " in dimension set " + dimset.getId()
175                     + " for dimension " + preferredDimension + ".");
176             return null;
177         } else {
178             switch (relatives.size()) {
179                 case 0: {
180                     _log.warn("No translation found for " + id + " in dimension set " + dimset.getId()
181                             + " for dimension " + preferredDimension + ".");
182                     // Note May 4, 2010 by Tony Field - this had been changed to
183                     // return the input ID but that
184                     // is incorrect. The contract clearly states that null is to
185                     // be returned if no matching
186                     // relatives are found. When null is returned and it is not
187                     // expected, often the incorrect
188                     // dimension filter is configured.
189                     return null;
190                 }
191                 case 1: {
192                     final AssetId relative = relatives.iterator().next();
193                     _log.trace("LocaleUtils.findTranslation: RELATIVE FOUND... " + relative.getType() + " '"
194                             + relative.getId() + "' // errno = " + ics.GetErrno());
195                     return relative;
196 
197                 }
198                 default: {
199                     throw new IllegalStateException("found more than one translation for asset " + id
200                             + " and that is not supposed to be possible.");
201                 }
202             }
203         }
204     }

209     private boolean _isInputAssetDimensionPreferred(final AssetId id, final long preferredDimension) {
210         final Dimension dim = getLocaleForAsset(id);
211         if (dim == null) {
212             return true; // if locale not found, tell that the asset is expected
213         }
214         // locale
215         return dim.getId().getId() == preferredDimension;
216     }

218     private static final PreparedStmt FIND_DIMSET_FOR_SITE_PREPAREDSTMT = new PreparedStmt(
219             "select ds.id as id from DimensionSet ds, Publication p, AssetPublication ap where p.name = ? and p.id = ap.pubid and ap.assetid = ds.id and ds.status != 'VO' order by ds.updateddate",
220             Arrays.asList("DimensionSet", "AssetPublication", "Publication"));
221 
222     static {
223         FIND_DIMSET_FOR_SITE_PREPAREDSTMT.setElement(0, "Publication", "name");
224     }

226     /**
227      * Locates a single dimension set in a site. If no match is found, an
228      * exception is thrown. If more than one match is found, an exception is
229      * thrown.
230      * 
231      * @param site site containing a dimension set
232      * @return DimensionSet ID
233      */
234     private long _locateDimensionSetForSite(final String site) {
235         if (StringUtils.isBlank(site)) {
236             throw new IllegalArgumentException("Required site name missing");
237         }
238         final StatementParam params = FIND_DIMSET_FOR_SITE_PREPAREDSTMT.newParam();
239         params.setString(0, site);
240         final IListIterable list = SqlHelper.select(ics, FIND_DIMSET_FOR_SITE_PREPAREDSTMT, params);
241 
242         final int numRows = list.size();
243         if (numRows == 0) {
244             throw new IllegalStateException(
245                     "A DimensionSet has not been defined for site '"
246                             + site
247                             + "'. Cannot determine any translation unless some locales (Dimensions) are enabled for that site. Aborting operation.");
248         }
249         if (numRows > 1) {
250             final StringBuilder msg = new StringBuilder("More than one dimension set found in site " + site
251                     + ".  Exactly one is expected.  Dimension set ids: ");
252             for (final Row row : list) {
253                 final String id = row.getString("id");
254                 LogDep.logDep(ics, "DimensionSet", id);
255                 msg.append(id).append(" ");
256             }
257             throw new IllegalStateException(msg.append(".").toString());
258         }
259 
260         final String id = list.iterator().next().getString("id");
261         LogDep.logDep(ics, "DimensionSet", id);
262         return Long.parseLong(id);
263     }

265     private DimensionFilterInstance _getPopulatedDimensionFilter(final DimensionSetInstance dimset,
266             final long dimensionId) {
267 
268         // Set the filter's preferred dimension
269         // Equivalent to:
270         // %><dimensionset:asset assettype="Dimension"
271         // assetid="<%=localeDimensionId%>" /><%
272         final Dimension preferredDimension = getDM().loadDimension(dimensionId);
273         if (preferredDimension == null) {
274             throw new RuntimeException("Attempted to load Dimension with id " + dimensionId + " but it came back null");
275         }
276         return _getPopulatedDimensionFilter(dimset, preferredDimension);
277     }

279     private DimensionFilterInstance _getPopulatedDimensionFilter(final DimensionSetInstance dimset,
280             final Dimension preferredDimension) {
281         DimensionFilterInstance filter;
282         try {
283             filter = dimset.getFilter();
284         } catch (final DimensionException e) {
285             throw new RuntimeException("Could not get Dimension Filter from DimensionSet", e);
286         }
287 
288         filter.setDimensonPreference(Collections.singletonList(preferredDimension));
289         return filter;
290     }

292     private DimensionSetInstance _getDimensionSet(final long dimSetId) {
293         final String DIMSET_OBJ_NAME = "LocaleUtils:findTranslation:theDimensionSet:DimensionSet";
294         // Load the site-specific DimensionSet asset
295         ics.SetObj(DIMSET_OBJ_NAME, null); // clear first
296         final FTValList args = new FTValList();
297         args.put("NAME", DIMSET_OBJ_NAME);
298         args.put("TYPE", "DimensionSet");
299         args.put("OBJECTID", Long.toString(dimSetId));
300         args.put("EDITABLE", "FALSE");
301         ics.runTag("ASSET.LOAD", args);
302 
303         if (ics.GetErrno() < 0) {
304             throw new IllegalStateException("Could not load dimension set.  Errno: " + ics.GetErrno());
305         }
306 
307         final Object o = ics.GetObj(DIMSET_OBJ_NAME);
308         ics.SetObj(DIMSET_OBJ_NAME, null);
309         if (o == null) {
310             throw new IllegalStateException("Could not load dimension set but we got no errno... unexpected...");
311         }
312 
313         DimensionSetInstance dimset;
314         if (o instanceof DimensionSetInstance) {
315             dimset = (DimensionSetInstance) o;
316         } else {
317             throw new IllegalStateException("Loaded an asset that is not a DimensionSetInstance.");
318         }
319         return dimset;
320     }

331         final String DIMSET_OBJ_NAME = "LocaleUtils:findTranslation:theDimensionSet:DimensionSet";
332         ics.SetObj(DIMSET_OBJ_NAME, null);
333         final AssetLoadByName a = new AssetLoadByName();
334         a.setAssetType("DimensionSet");
335         a.setAssetName(name);
336         a.setEditable(false);
337         a.setName(DIMSET_OBJ_NAME);
338         a.execute(ics);
339 
340         if (ics.GetErrno() < 0) {
341             throw new IllegalStateException("Could not load dimension set.  Errno: " + ics.GetErrno());
342         }
343 
344         final Object o = ics.GetObj(DIMSET_OBJ_NAME);
345         ics.SetObj(DIMSET_OBJ_NAME, null);
