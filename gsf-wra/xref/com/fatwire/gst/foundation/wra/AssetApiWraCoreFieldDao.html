174     public String resolveSite(String c, String cid) {
175         final StatementParam param = AP_STMT.newParam();
176         param.setString(0, c);
177         param.setLong(1, Long.parseLong(cid));
178         String result = null;
179         for (Row pubid : SqlHelper.select(ics, AP_STMT, param)) {
180             if (result != null) {
181                 LOG.warn("Found asset "
182                         + c
183                         + ":"
184                         + cid
185                         + " in more than one publication. It should not be shared; aliases are to be used for cross-site sharing.  Controller will use first site found: "
186                         + result);
187             } else {
188                 result = pubid.getString("name");
189             }
190         }
191         return result;
192     }

194     static final PreparedStmt FIND_P = new PreparedStmt(
195             "SELECT art.oid FROM AssetRelationTree art, AssetPublication ap, Publication p WHERE ap.assetid = art.oid "
196                     + "AND ap.assettype = 'Page' AND ap.pubid = p.id AND p.name = ? "
197                     + "AND art.otype = ap.assettype AND art.nid in "
198                     + "(SELECT nparentid FROM AssetRelationtree WHERE otype=? AND oid=? AND ncode='-') ORDER BY ap.id",
199             Arrays.asList("AssetRelationTree", "AssetPublication", "Publication"));
200 
201     static {
202         FIND_P.setElement(0, "Publication", "name");
203         FIND_P.setElement(1, "AssetRelationTree", "otype");
204         FIND_P.setElement(2, "AssetRelationTree", "oid");
205     }

207     /**
208      * Locate the page that contains the specified Web-Referenceable Asset.
209      * <p/>
210      * A WRA is supposed to just be placed on one page (in the unnamed
211      * association block), and this method locates it. If it is not found, 0L is
212      * returned.
213      * <p/>
214      * If multiple matches are found, a warning is logged and the first one is
215      * returned.
216      * 
217      * @param wraAssetIdWithSite id of the web-referenceable asset. Site is
218      *            included
219      * @return page asset ID or 0L.
220      */
221     public long findP(AssetIdWithSite wraAssetIdWithSite) {
222         ics.ClearErrno();
223         final StatementParam param = FIND_P.newParam();
224         param.setString(0, wraAssetIdWithSite.getSite());
225         param.setString(1, wraAssetIdWithSite.getType());
226         param.setLong(2, wraAssetIdWithSite.getId());
227         long result = 0L;
228         for (final Row r : SqlHelper.select(ics, FIND_P, param)) {
229             if (result != 0L)
230                 LOG.warn("Asset "
231                         + wraAssetIdWithSite
232                         + " was found as the primary content on multiple pages in the site.  Web-referenceable assets should only be the primary content on one Page to comply with SEO rules.  Automatically selecting the oldest page");
233             else
234                 result = r.getLong("oid");
235         }
236         if (result == 0L) {
237             // could not locate
238         }
239         return result;
240 
241     }
